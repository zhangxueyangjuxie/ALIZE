193 // Compute the entropy from an Histo
194 double computeEntropy(Histo & hist)
195 {
196  // integral is done by summing with a step (weighted by log(x) function)
197  double prob=0.0;
198  double entropy=0.0;
199  double bound=hist.lowerBound(0);
200  double step=0.001;
201  while (bound < hist.higherBound(hist.size()-1)) {
202  prob=hist(bound); // get density
203  if (prob<1e-20) {entropy+=0.0;} // take care of lim x->0 x*log(x) (=0)
204  else {entropy+=-prob*step*log(prob);} //get the area of the bin to get P(x), indeed P(x)=sum_x(p(x))
205  bound+=step;
206  //cerr << "prob: "<<prob<<" entropy: "<<entropy<<endl;
207  }
208 return entropy;
209 }

212 // Compute mean from an Histo
213 double computeMean(Histo & hist)
214 {
215  // integral is done by summing with a step (weighted by log(x) function)
216  double prob=0.0;
217  double mean=0.0;
218  double bound=hist.lowerBound(0);
219  double step=0.001;
220  while (bound < hist.higherBound(hist.size()-1)) {
221  prob=step*hist(bound);
222  mean+=bound*prob; //get the area of the bin to get P(x), indeed P(x)=sum_x(p(x))
223  bound+=step;
224  //cerr << "prob: "<<prob<<" mean: "<<mean<<endl;
225  }
226 return mean;
227 }

816 double likelihoodGD(const DistribGD& d,const DistribGD &m){
