1691 // Platt's binary SVM Probablistic Output: an improvement from Lin et al.
1692 static void sigmoid_train(
1693  int l, const double *dec_values, const double *labels,
1694  double& A, double& B)
1695 {
1696  double prior1=0, prior0 = 0;
1697  int i;
1698 
1699  for (i=0;i<l;i++)
1700  if (labels[i] > 0) prior1+=1;
1701  else prior0+=1;
1702 
1703  int max_iter=100; // Maximal number of iterations
1704  double min_step=1e-10; // Minimal step taken in line search
1705  double sigma=1e-12; // For numerically strict PD of Hessian
1706  double eps=1e-5;
1707  double hiTarget=(prior1+1.0)/(prior1+2.0);
1708  double loTarget=1/(prior0+2.0);
1709  double *t=Malloc(double,l);
1710  double fApB,p,q,h11,h22,h21,g1,g2,det,dA,dB,gd,stepsize;
1711  double newA,newB,newf,d1,d2;
1712  int iter;
1713 
1714  // Initial Point and Initial Fun Value
1715  A=0.0; B=log((prior0+1.0)/(prior1+1.0));
1716  double fval = 0.0;
1717 
1718  for (i=0;i<l;i++)
1719  {
1720  if (labels[i]>0) t[i]=hiTarget;
1721  else t[i]=loTarget;
1722  fApB = dec_values[i]*A+B;
1723  if (fApB>=0)
1724  fval += t[i]*fApB + log(1+exp(-fApB));
1725  else
1726  fval += (t[i] - 1)*fApB +log(1+exp(fApB));
1727  }
1728  for (iter=0;iter<max_iter;iter++)
1729  {
1730  // Update Gradient and Hessian (use H' = H + sigma I)
1731  h11=sigma; // numerically ensures strict PD
1732  h22=sigma;
1733  h21=0.0;g1=0.0;g2=0.0;
1734  for (i=0;i<l;i++)
1735  {
1736  fApB = dec_values[i]*A+B;
1737  if (fApB >= 0)
1738  {
1739  p=exp(-fApB)/(1.0+exp(-fApB));
1740  q=1.0/(1.0+exp(-fApB));
1741  }
1742  else
1743  {
1744  p=1.0/(1.0+exp(fApB));
1745  q=exp(fApB)/(1.0+exp(fApB));
1746  }
1747  d2=p*q;
1748  h11+=dec_values[i]*dec_values[i]*d2;
1749  h22+=d2;
1750  h21+=dec_values[i]*d2;
1751  d1=t[i]-p;
1752  g1+=dec_values[i]*d1;
1753  g2+=d1;
1754  }
1755 
1756  // Stopping Criteria
1757  if (fabs(g1)<eps && fabs(g2)<eps)
1758  break;
1759 
1760  // Finding Newton direction: -inv(H') * g
1761  det=h11*h22-h21*h21;
1762  dA=-(h22*g1 - h21 * g2) / det;
1763  dB=-(-h21*g1+ h11 * g2) / det;
1764  gd=g1*dA+g2*dB;
1765 
1766 
1767  stepsize = 1; // Line Search
1768  while (stepsize >= min_step)
1769  {
1770  newA = A + stepsize * dA;
1771  newB = B + stepsize * dB;
1772 
1773  // New function value
1774  newf = 0.0;
1775  for (i=0;i<l;i++)
1776  {
1777  fApB = dec_values[i]*newA+newB;
1778  if (fApB >= 0)
1779  newf += t[i]*fApB + log(1+exp(-fApB));
1780  else
1781  newf += (t[i] - 1)*fApB +log(1+exp(fApB));
1782  }
1783  // Check sufficient decrease
1784  if (newf<fval+0.0001*stepsize*gd)
1785  {
1786  A=newA;B=newB;fval=newf;
1787  break;
1788  }
1789  else
1790  stepsize = stepsize / 2.0;
1791  }
1792 
1793  if (stepsize < min_step)
1794  {
1795  info("Line search fails in two-class probability estimates\n");
1796  break;
1797  }
1798  }
1799 
1800  if (iter>=max_iter)
1801  info("Reaching maximal iterations in two-class probability estimates\n");
1802  free(t);
1803 }

1815 static void multiclass_probability(int k, double **r, double *p)
1816 {
1817  int t,j;

1820  double *Qp=Malloc(double,k);
1821  double pQp, eps=0.005/k;
1822 
1823  for (t=0;t<k;t++)
1824  {
1825  p[t]=1.0/k; // Valid if k = 1

1828  for (j=0;j<t;j++)

1836  Q[t][j]=-r[j][t]*r[t][j];
1837  }
1838  }
1839  for (iter=0;iter<max_iter;iter++)
1840  {
1841  // stopping condition, recalculate QP,pQP for numerical accuracy
1842  pQp=0;

1845  Qp[t]=0;

1852  {

1854  if (error>max_error)
1855  max_error=error;
1856  }

1858

1860  {

1867  p[j]/=(1+diff);
1868  }
1869  }
1870  }
1871  if (iter>=max_iter)
1872  info("Exceeds max_iter in multiclass_prob\n");

1875  free(Qp);
