416  actualViterbiProb.clear();
417  previousViterbiProb.clear();

419  previousViterbiProb.setSize(cluster.getCount());

421  actualViterbiProb[iprob]=0;

451  else {
452  cout << "Unknown train Algo ! "<< endl;
453  exit(-1);
454  }
455 
456  segTemp=actualSeg;
457 
458  if(testAlgo == "Viterbi")
459  viterbiDecoding(config,actualHMM,cluster,actualSeg,ss,fs,labelServer,actualViterbiProb);

461  cout << "Unknown test Algo ! "<< endl;

470 
471 
472  }while((!isStopCriterionReached(config,actualHMM,previousHMM,actualSeg,previousSeg,alreadyUsedSeg,ss,fs,initSpeakerTime)) && (actualSeg.getClusterCount()<limitSpeaker));// segmentation loop
473 }//if isSegAvai
