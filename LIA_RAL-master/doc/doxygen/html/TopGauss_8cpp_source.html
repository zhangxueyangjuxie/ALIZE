137  StatServer ss(config);
138  MixtureGDStat &acc=ss.createAndStoreMixtureStat(UBM);
139  unsigned long _mixsize=UBM.getDistribCount();
140  String labelSelectedFrames =config.getParam("labelSelectedFrames");
141  unsigned long begin=fs.getFirstFeatureIndexOfASource(featureFilename);
142  fs.seekFeature(begin);
143  SegServer segmentsServer;
144  LabelServer labelServer;
145  initializeClusters(featureFilename,segmentsServer,labelServer,config);
146  verifyClusterFile(segmentsServer,fs,config);
147  unsigned long codeSelectedFrame=labelServer.getLabelIndexByString(labelSelectedFrames);
148  SegCluster& selectedSegments=segmentsServer.getCluster(codeSelectedFrame);
149  acc.resetLLK();
150  double topD=config.getParam("topGauss").toDouble();
151  if (verbose) {if(topD<1.0) cout << "LLK %="<< topD << "% ";else cout << "Top-"<<topD<<" ";}
152 
153  // Class values
154  _nt=totalFrame(selectedSegments);
155  _nbg.setSize(_nt); _idx.setSize(0);_snsw.setSize(0); _snsl.setSize(0);
156  _nbg.setAllValues(0); _idx.setAllValues(0);_snsw.setAllValues(0.0);_snsl.setAllValues(0.0);
157  _nbgcnt=0;
158  Seg *seg; // current selected segment
159  selectedSegments.rewind();
160  unsigned long t=0; //cnt frames
161  while((seg=selectedSegments.getSeg())!=NULL){
162  unsigned long begin=seg->begin()+fs.getFirstFeatureIndexOfASource(seg->sourceName());
163  fs.seekFeature(begin);
164  Feature f;
165  for (unsigned long idxFrame=0;idxFrame<seg->length();idxFrame++){
166  fs.readFeature(f);
167  double llk=acc.computeAndAccumulateLLK(f,1.0,DETERMINE_TOP_DISTRIBS);
168  const LKVector &topV=ss.getTopDistribIndexVector();
169  double lk_tot=exp(llk);
170 
171  double val=0.0;
172  if (topD<1.0) {
173  for(unsigned long j=0;j<_mixsize;j++){
174  if (val > topD*lk_tot) break;
175  val+=(topV[j].lk);
176  _nbg[t]++;
177  }
178  } else _nbg[t]=(unsigned long)topD;
179  _nbgcnt+=_nbg[t];
180 
181  double snsw=1.0;
182  double snsl=lk_tot;
183  for(unsigned long j=0;j<_nbg[t];j++) {
184  _idx.addValue(topV[j].idx);
185  snsw -=UBM.weight(topV[j].idx);
186  snsl -=topV[j].lk;
187  }
188 
189  _snsw.addValue(snsw);
190  if (snsl < EPS_LK)
191  _snsl.addValue(EPS_LK);
192  else _snsl.addValue(snsl);
193  t++;
194  }
195  }
196  if (t!=_nt) cout << "W: t("<<t<<") != _nt(" <<_nt<<")"<<endl;
197 return acc.getMeanLLK();
198 }
