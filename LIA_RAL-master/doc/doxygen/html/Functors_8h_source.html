533 // linear access for packet ops:
534 // 1) initialization
535 // base = [low, ..., low] + ([step, ..., step] * [-size, ..., 0])
536 // 2) each step
537 // base += [size*step, ..., size*step]
538 template <typename Scalar>
539 struct linspaced_op_impl<Scalar,false>
540 {
541  typedef typename packet_traits<Scalar>::type Packet;
542 
543  linspaced_op_impl(Scalar low, Scalar step) :
544  m_low(low), m_step(step),
545  m_packetStep(pset1<Packet>(packet_traits<Scalar>::size*step)),
546  m_base(padd(pset1<Packet>(low),pmul(pset1<Packet>(step),plset<Scalar>(-packet_traits<Scalar>::size)))) {}
547 
548  template<typename Index>
549  EIGEN_STRONG_INLINE const Scalar operator() (Index i) const { return m_low+i*m_step; }
550  template<typename Index>
551  EIGEN_STRONG_INLINE const Packet packetOp(Index) const { return m_base = padd(m_base,m_packetStep); }
552 
553  const Scalar m_low;
554  const Scalar m_step;
555  const Packet m_packetStep;
556  mutable Packet m_base;
557 };
