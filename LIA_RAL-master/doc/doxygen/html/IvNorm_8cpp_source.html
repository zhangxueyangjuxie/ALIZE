105  test.sphericalNuisanceNormalization(config);

108  String ldaFilename = config.getParam("matrixFilesPath")+config.getParam("ldaMatrix")+config.getParam("loadMatrixFilesExtension");

110  test.rotateLeft(LDAmat);
111  }
112 
113  //Save test data

115 
116  //if input is a list of files, then save only test segments and not models (they are the same)
117  if(config.existsParam("inputVectorFilename") && !config.existsParam("targetIdList"))
118  test.saveSegments(outputDir, config);
119 
120  //if there are models and test, then save everything
