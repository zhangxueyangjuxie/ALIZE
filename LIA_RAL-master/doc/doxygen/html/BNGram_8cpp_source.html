70 BNgramNode * BNgram::_newNode(short int symb,BNgramNode*br=NULL){
71  BNgramNode * ptr=new BNgramNode;
72  ptr->child=new BNgramNode* [_nbElt];
73  ptr->count=new unsigned long [_nbElt];
74  for (short int i=0;i<_nbElt;i++){ptr->child[i]=NULL;ptr->count[i]=0;}
75  ptr->idx=(symb/_nbElt)*_nbElt;
76  ptr->brother=br;
77  return ptr;
78 }
79 BNgramNode * BNgram::_findInsert(BNgramNode* ptr,short int idx){
80  // TAKE CARE, A LIST ALWAYS BEGIN BY A NODE WITH IDX=0
81  if (idx<(ptr->idx+_nbElt))return ptr; // current node is the good one
82  while ((ptr->brother!=NULL) && (idx>=(ptr->brother->idx+_nbElt))) ptr=ptr->brother;
83  if (ptr->brother==NULL) {ptr->brother=_newNode(idx); return ptr->brother;} // new node at the end
84  if ((idx>=ptr->brother->idx)&& (idx<ptr->brother->idx+_nbElt)) return ptr->brother; // the node exists
85  ptr->brother=_newNode(idx,ptr->brother); return ptr->brother; // new node in the midle;
86 }
87 void BNgram::_add(short int symb,bool first,short int nb=1){
88  // TAKE CARE, A LIST ALWAYS BEGIN BY A NODE WITH IDX=0
89  // 1: seed memorizes the good seed for the n level. It is set and created if necessary by the level n-1
90  static BNgramNode* seed;
91  static short int level; // level is the current ngram order set by the level n-1 (i.e. at the end of this function)
92  if (first) {
93  seed=_seed; // If first level (unigram), begin at the Btree seed, else searchSeed and level are ready
94  level=0;
95  _totalCount++;
96  }
97  if (debug) cout <<"add symb["<<symb<<"] Level["<<level<<"]"<<endl;
98  // 2: find the good node and add the count
99  seed=_findInsert(seed,symb); // find the good node, create it if necessary (3 takes care of the begin of the list)
100  seed->count[symb%_nbElt]+=nb;
101 
102  //3: Now, prepare the work for the next level
103  if (seed->child[symb%_nbElt]==NULL) seed->child[symb%_nbElt]=_newNode(0);
104  seed=seed->child[symb%_nbElt];
105  level++;
106 }
