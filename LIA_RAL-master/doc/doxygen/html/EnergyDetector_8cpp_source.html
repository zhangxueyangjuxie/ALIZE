106 double computeEnergyThreshold(FeatureServer & fs,double pSelect,unsigned long nbBins=100)
107 {
108  Histo histo(nbBins); // Create an histo accumulator with 100 bins
109  Feature f; // reset the reader at the begin of the input stream
110  fs.reset(); // feature server reset
111 for (unsigned long ind=0;fs.readFeature(f); ind++) // feature loop
112  histo.accumulateValue(f[0]); // Accumulate the energy in the histo Accumulator
113  histo.computeHisto(); // Compute the histo
114  long i=nbBins-1;
115  real_t count=0;
116 while((i>=0) && (count<=pSelect)){ // Find the bin corresponding to the percentage of data wanted
117  count+=histo.count(i)*(histo.higherBound(i)-histo.lowerBound(i));
118  i--;
119  }
120  double threshold;
121  if (i>=0) threshold=histo.higherBound(i); // Set the threshold to the higherBound of the next bin
122  else threshold=histo.lowerBound(0); // if 100% of data should be selected
123  if (verbose) cout << "Percentage wanted["<<(int) (pSelect*100.0) <<"]Energy threshold["<<threshold<<"]"<<endl;
124  return threshold;
125 }

128 unsigned long selectFrames(FeatureServer &fs,SegServer & segServer,double threshold,SegCluster &selectedSeg,SegCluster &outputSeg,String labelOutput,String fileName)
129 {
130  unsigned long countFrames=0;
131  fs.reset(); // feature server reset
132  unsigned long ind=0;
133  unsigned long begin=0;
134  bool in=false;
135  Seg *seg; // current selectd segment
136  selectedSeg.rewind(); // reset the reader at the begin of the input stream
137  while((seg=selectedSeg.getSeg())!=NULL){ // For each input segments
138  for (unsigned long idx=seg->begin();idx<seg->begin()+seg->length();idx++){ // for each frame
139  Feature f;
140  fs.seekFeature(idx);
141  fs.readFeature(f);
142  if (f[0]>threshold){ // the frame is selected
143  countFrames++;
144  if (in==false){ // Begin of a new segment
145  in=true;
146  begin=ind;
147  }
148  }
149  else if (in){ // End of a segment
150  in=false;
151  Seg & segFake=segServer.createSeg(begin,ind-begin,0, // Create a segment - Take care : length=end-begin+1 but ind =end+1 !!
152  labelOutput,fileName);
153  outputSeg.add(segFake); // Add a segment
154  }
155  ind++; // Increment the frame index
156  } // end of one input segment
157  if (in){ // deal with the last energized segmeent inside the current input segment
158  in=false;
159  Seg & segFake=segServer.createSeg(begin,ind-begin+1,0, // Create a segment
160  labelOutput,fileName);
161  outputSeg.add(segFake); // Add a segment - Take care : length=end-begin+1 and ind=end in this case !!
162  }
163  } // end feature loop
164 
165  return countFrames;
166 }

173 MixtureGD &energyMixtureInit(MixtureServer &ms, StatServer &ss, FeatureServer &fs,MixtureGD
174 &world,SegCluster &selectedSegments,const DoubleVector &globalCov, Config& config)
175 {
176  unsigned long vectSize = world.getVectSize();
177  unsigned long distribCount = world.getDistribCount();
178 
179  double mean=-2.0;
180  double meanIncrement;
181  if (distribCount>1) meanIncrement=4.0/(double)(distribCount-1);
182  else meanIncrement=1;
183  for (unsigned long indg=0;indg<distribCount;indg++) { // For each component
184  DistribGD& d = world.getDistrib(indg); // Get it
185  for (unsigned long c=0;c<vectSize;c++,mean+=meanIncrement){ // copy it
186  d.setCov(1.0, c);
187  d.setMean(mean, c);
188  }
189  d.computeAll();
190  }
191  world.equalizeWeights(); // set weight = 1/distribCount for each distrib
192  return world;
193 }
