146 void WindowLLR::dec(unsigned long idxFrame){
147  if (_count<_size){ //window is not full
148  _count++;
149  unsigned long eIdx=(_bIdx+_count-1)%_size;
150  (*_idxA)[eIdx]=idxFrame;
151  }
152  else{// window is full, real dec (shift the window, step _dec frame)
153  for (unsigned long wIdx=0;wIdx<_dec;wIdx++){
154  //suppress the begin value
155  for (unsigned long cIdx=0;cIdx<_nClient;cIdx++)
156  (*_accLlrA)[cIdx]-=(*_llrM)(_bIdx,cIdx);
157  _bIdx=(_bIdx+1)%_size;
158  }
159  _count-=(_dec-1);
160  (*_idxA)[(_bIdx+_count-1)%_size]=idxFrame;
161  }
162 }
