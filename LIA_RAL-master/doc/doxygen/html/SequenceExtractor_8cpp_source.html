154 unsigned long CommonPartTree::_findMaxSeq(CommonPartTreeNode *ptr,Seq &seq, short int & order){
155  if (ptr==NULL){ order=0; return 0;}
156  short int orderCh=order+1;
157  short int orderBr=order;
158  Seq seqBr=seq;
159  seq.setLength(order+1);seq[order].setSymb(ptr->symb);
160  unsigned long brCount=_findMaxSeq(ptr->br,seqBr,orderBr);
161  unsigned long chCount=_findMaxSeq(ptr->ch,seq,orderCh);
162 
163  if (orderBr<orderCh){seq[order].setSymb(ptr->symb); order=orderCh;return chCount;} // child is the longuest sequence and win
164  // child and/or borther == NULL
165  if (orderCh==0){
166  if (orderBr==0) return ptr->count; // Brother==NULL, current is the winner
167  if (orderBr>order) {order=orderBr; seq=seqBr;return brCount;} // Brother find the longest sequence
168  // equal length and child==NULL
169  if (ptr->count>brCount) return ptr->count; // current is the winner
170  else {order=orderBr; seq=seqBr;return brCount;} // Brother is the winner
171  }
172  // child and Brother !=NULL
173  if (orderBr>orderCh){order=orderBr; seq=seqBr;return brCount;} // Brother find the longest sequence and win
174  // Equal length (and child and brother !=NULL)
175  if (brCount>(chCount)){order=orderBr;seq=seqBr;return brCount;} // Brother has the max count and win
176  else{seq[order].setSymb(ptr->symb);order=orderCh;return chCount;} // child has the max count and win
177 }

198 // Take care, a sequence could contain more than one symbol by node...
199 CommonPartTreeNode* CommonPartTree::_suppressSeq(CommonPartTreeNode * ptr,Seq &seq,
200  short int order, unsigned long &childCountDelta){
201  if (order>=seq.getLength())
202  throw Exception("sequence longer than the tree" , __FILE__, __LINE__);
203  CommonPartTreeNode* ptrRet=ptr;
204  CommonPartTreeNode* ptrTmp=ptr;
205  while ((ptr!=NULL)&& (!seq[order].isIn(ptr->symb))) {ptrTmp=ptr;ptr=ptr->br;}
206  if (seq[order].isIn(ptr->symb)){ // We are in the good tree branch
207  if (order==seq.getLength()-1) { // and at the end of the sequence
208  childCountDelta=ptr->count;
209  ptrTmp->br=ptr->br;
210  if (ptrRet==ptr) ptrRet=ptr->br;
211  delete ptr;
212  }
213  else{ // We are in the good branch but not at the end of the sequence
214  ptr->ch=_suppressSeq(ptr->ch,seq,order+1,childCountDelta);
215  if (ptr->count<childCountDelta) throw Exception("count problem in the tree, childcount < delta" , __FILE__, __LINE__);
216  ptr->totalChildCount-=childCountDelta;
217  ptr->count-=childCountDelta;
218  if (ptr->count==0) { // We are in a node with a new count=0, to be destroyed
219  ptrTmp->br=ptr->br;
220  if (ptrRet==ptr) ptrRet=ptr->br;
221  }
222  }
223  }
224  else{ // Branch nor find
225  childCountDelta=0;
226  ptrRet= NULL;
227  }
228  return ptrRet;
229 }

724 // sequence extractor builds a common part tree defining a set of variable length symbols with the as same
725 // as possible probabilities to occur.
726 // the input data are the grams, from 1 to maxOrder. Each contains the observed ngram with the corresponding probability
727 // the required number of symbols (nbSym) should also be defined a priori. It gives pSym=1/nbSym, the target probability for each symbol
728 // The algo consists in selecting the maximum length sequence (S0,S1,..,Sn) with the maximum probability and to aglomerate it with common
729 // part sequences until pSym is reached. I.E, making a symbol by combining (a,b,c,d) with (a,b,c,e) (a,b,c,f)...
730 // The variable length symbol is then inserted in a common part tree: the result !
731 // the algo is repeated until the number of symbol is reached
732 int sequenceExtractor(Config& config){
733 
734  int maxOrder=config.getParam("maxOrder").toLong(); // Getting the order of ngrams. Ngrams from 1 to maxOrder should be present
735  unsigned long maxNgram=config.getParam("maxNgram").toLong(); // Max ngram by file
736  int nbSymb=config.getParam("nbInputSymb").toLong(); // Number of input symbols
737  const int nbSeq=config.getParam("nbOutputSymb").toLong(); // Number of sequence to detect
738  bool equalInputInfo=false;
739  if (config.existsParam("equalInputInfo")) equalInputInfo=config.getParam("equalInputInfo").toBool(); // Work with count*size of the sequence
740  String ngramFilename=config.getParam("ngramFilename"); // Base filename for the ngram. No order filename have the form ngramFilenameN.ext
741  String ngramExt=config.getParam("ngramExt"); // Ext for the ngram files, should include the .
742  String outputFilenameInfo=config.getParam("outputInfoFilename"); // The complete output filename for info file
743  ofstream outputFileInfo(outputFilenameInfo.c_str(),ios::out| ios::trunc);
744  String outputFilename=config.getParam("outputFilename"); // The complete output filename for the decoder tree
745  ofstream outputFile(outputFilename.c_str(),ios::out| ios::trunc);
746 
747  CommonPartTree initialTree; // The tree with all the info
748  if (verbose) cout << "reading the ngram files"<<endl;
749  for (int order=1;order<=maxOrder;order++){ // Load the ngrams, put it in a common part tree
750  NGram nGram(order,maxNgram);
751  nGram.load(ngramFilename+String::valueOf(order)+ngramExt,config);
752  if (debug) nGram.showTable();
753  initialTree.addNGram(nGram); // insert all the ngram in the tree
754  }
755  initialTree.setTotalCount(initialTree.getTotalChildCount()); // All the symbols are seen in the unigrams
756  if (verbose) cout <<"total input symbols:"<<initialTree.getTotalChildCount()<<endl;
757  if (debug) {cout<<"initial tree"<<endl; initialTree.show();}
758 
759  // Initialize the tree
760  SequenceDecoder seqTree(nbSymb);
761  Seq currentSeq(maxOrder,nbSymb);
762  // Begin the main loop
763  unsigned long remainingInputSymb=initialTree.getTotalChildCount();
764  bool fEnd=false;
765  for (int seq=0;(seq<nbSeq) && (!fEnd);seq++){ // Main sequence loop - a sequence is decided at each iteration
766  if (verbose) cout << "begin of the main sequence detection loop, seq:"<<seq<<endl;
767  unsigned long targetSeqCount=remainingInputSymb/(nbSeq-seq); // Count for each variable length symbol
768  if (verbose) cout << "totalchildcount"<<initialTree.getTotalChildCount()<<" Remaining input symb:"<<remainingInputSymb<<endl;
769  if (verbose) cout <<"target Seq count["<<targetSeqCount<<"]"<<endl;
770  currentSeq.init(); // reset the current seq
771  unsigned long seqCount=initialTree.findMaxSeq(currentSeq); // Finding the maximum length sequence with the max count
772  if (equalInputInfo) seqCount*=currentSeq.getLength(); // If you want to take into account the number of input symbols and not only the count
773  if (currentSeq.getLength()==0) fEnd=true;
774  else{
775  if (verbose){
776  cout <<"First Seq["<<seq<<"] Length["<<currentSeq.getLength();
777  if (equalInputInfo)cout<<"] Seq input symbols count[";
778  else cout <<"] SeqCount[";
779  cout<<seqCount<<"]"<<endl;currentSeq.show();
780  }
781  initialTree.suppressSeq(currentSeq); // suppress the sequence in the initialTree
782  if (debug){ cout <<"tree after suppress"<<endl; initialTree.show();}
783  seqTree.addSequence(currentSeq,seq);
784  if (debug) {cout <<"seq tree after adding initial sequence"<<endl;seqTree.show();}
785  short int length=currentSeq.getLength()-1;// -1 because we want to suppress the last level
786  while ((seqCount<targetSeqCount)&&(length>=0)){
787  bool end=false;
788  while ((!end)&&(length>=0)&&(seqCount<targetSeqCount)){
789  currentSeq.init(length); // withdrath the info of the end of the sequence
790  if (debug) {cout <<"current path"<<endl;currentSeq.show();}
791  unsigned long deltaSeqCount=initialTree.findMaxEndSeq(currentSeq);
792  if (equalInputInfo) deltaSeqCount*=currentSeq.getLength(); // If you want to take into account the number of input symbols and not only the count
793  end=(deltaSeqCount==0)||(currentSeq.getLength()==0);
794  if (!end){
795  seqCount+=deltaSeqCount;
796  if (verbose){
797  cout <<"Seq["<<seq<<"] add new seq Length["<<currentSeq.getLength();
798  if (equalInputInfo)cout<<"] Seq input symbols count[";
799  else cout <<"] SeqCount[";
800  cout<<seqCount<<"]"<<endl;currentSeq.show();
801  }
802  initialTree.suppressSeq(currentSeq); // suppress the sequence in the initialTree
803  if (debug){ cout <<"tree after suppress seq"<<endl; initialTree.show();}
804  seqTree.addSequence(currentSeq,seq);
805  if (debug){cout <<"seq tree after adding seq"<<endl;seqTree.show();}
806  length=currentSeq.getLength()-1;
807  }
808  else length--;
809  }
810  }
811  remainingInputSymb-=seqCount;
812  if (seqCount==0) fEnd=true;
813  else{
814  outputFileInfo<<"Sequence["<<seq<<"] Count["<<seqCount<<"]"<<endl;
815  seqTree.setNbOutputSeq(seq+1);
816  }
817  }
818  }
819 
820  if (verboseLevel>1){cout <<"FINAL SEQUENCE TREE"<<endl; seqTree.show();}
821  if (verbose) seqTree.toFile(outputFileInfo);
822  seqTree.save(outputFile);
823 
824  return 0;
825 }
