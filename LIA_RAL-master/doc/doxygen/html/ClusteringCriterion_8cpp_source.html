127 /**********************************************************
128 * bicCrit: Application of BIC criterion for clustering
129 ***********************************************************/
130 double bicCrit(Config& config, SegCluster& c1, SegCluster &c2, SegCluster &c12, StatServer& ss, FeatureServer& fs, MixtureGD& m1, MixtureGD& m2, MixtureGD& m12){
131 
132 double gllr = gllrCrit(config, c1, c2, c12, ss, fs, m1, m2, m12);
133 double lambda = 1.0;
134 double P = 0.5 * ((2 * fs.getVectSize()+1)*m1.getDistribCount()) * log((double)totalFrame(c1)+totalFrame(c2));
135 double bic = (-1 * gllr) - lambda * P;
136 if (verbose) cout << "BIC ("<<totalFrame(c1)<<"/"<<totalFrame(c2)<< ") => " << bic << " P => " << P << endl;
137 
138 return bic;
139 }
