108 void projectOnSubSpace(RealVector<double> &x, Matrix<double> &U, RealVector <double>&proj) { //fast version
109  // trick is to transform x'=S'Sx in x'=(U(U'x))' so that no need to compute U explicitely
110  double *_u=U.getArray();
111  RealVector <double> tmp;
112  tmp.setSize(U.rows());
113  tmp.setAllValues(0.0);
114  proj.setAllValues(0.0);
115  if (verboseLevel > 1) cout<<"x: ("<<x.size()<<") - U; ("<<U.rows()<<","<<U.cols()<<")"<<endl;
116  for (unsigned long i=0;i<U.rows();i++)
117  for (unsigned long j=0;j<U.cols();j++)
118  tmp[i]+=_u[i*U.cols()+j]*x[j];
119  U.transpose();
120  _u=U.getArray();
121  for (unsigned long i=0;i<U.rows();i++)
122  for (unsigned long j=0;j<U.cols();j++)
123  proj[i]+=_u[i*U.cols()+j]*tmp[j];

125 }

128 void computeNap(MixtureGD &M,Matrix <double> &U) {
129  unsigned long svSize=M.getDistribCount()*M.getVectSize();
130  RealVector <double> v(svSize,svSize);
131  v.setAllValues(0.0);
132  RealVector <double> proj(svSize,svSize);
133  proj.setAllValues(0.0);
134  modelToSv(M,v);
135  projectOnSubSpace(v,U,proj);
136  v-=proj;
137  svToModel(v,M);
138 }

148  unsigned long svSize=max(U.rows(),U.cols());

153  RealVector <double> channel(svSize,svSize);
154  modelToSv(T,t);
155  modelToSv(M,m);
156  double normT=0.0;
157  double angle=0.0;
158  for (unsigned long i=0;i<t.size();i++) // norm du test
159  normT+=t[i]*t[i];

164  proj[i]=(angle/normT)*t[i];
165  projectOnSubSpace(m,U,tmp); // Channel effect for test
