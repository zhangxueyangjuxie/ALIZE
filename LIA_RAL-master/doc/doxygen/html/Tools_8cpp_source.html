327 /****************************************************************
328 * computeTransitions: HMM transition computation:
329 * either equiprobable or with a constant pii and pij ratio
330 *****************************************************************/
331 void computeTransitions(Config& config,hmm& actualHMM,String transitionMethod,double gamma,SegServer& Server){
332 
333 if(transitionMethod=="Equiprob"){
334  if((gamma<0)||(gamma>1)){
335  cout<<"Gamma is not in interval 0 1! "<<gamma<<endl;
336  exit(-1);
337  }
338  for(unsigned long i=0;i<actualHMM.getNbState();i++){
339  for(unsigned long j=0;j<actualHMM.getNbState();j++){
340  if(i==j){
341  actualHMM.setTransition(gamma,i,j);
342  if ((verbose) && (verboseLevel == 2)) cout<<gamma<<" ";
343  }
344  else{
345  actualHMM.setTransition((1.0-gamma)/(double)(actualHMM.getNbState()-1.0),i,j);
346  if ((verbose) && (verboseLevel == 2)) cout<<(1.0-gamma)/(double)(actualHMM.getNbState()-1.0)<<" ";
347  }
348  }//for j
349  if ((verbose) && (verboseLevel == 2)) cout<<endl;
350  }//for i
351 }//if equiprob
352 if(transitionMethod=="RapConst"){
353  double transij = ((double)1.0 / (gamma + (double)(actualHMM.getNbState()-1.0)));
354  double transii = (gamma * transij);
355  for(unsigned long i=0;i<actualHMM.getNbState();i++){
356  for(unsigned long j=0;j<actualHMM.getNbState();j++){
357  if(i==j){
358  actualHMM.setTransition(transii,i,j);
359  if ((verbose) && (verboseLevel == 2)) cout<<transii<<" ";
360  }
361  else{
362  actualHMM.setTransition(transij,i,j);
363  if (verbose) cout<<transij<<" ";
364  }
365  }//for j
366  if ((verbose) && (verboseLevel == 2)) cout<<endl;
367  }//for i
368 }//if RapConst
369 
370 
371 if(transitionMethod=="Unity"){
372  for(unsigned long i=0;i<actualHMM.getNbState();i++){
373  for(unsigned long j=0;j<actualHMM.getNbState();j++){
374  actualHMM.setTransition(1.0,i,j);
375  if ((verbose) && (verboseLevel == 2)) cout<<1.0<<" ";
376  }//for j
377  if ((verbose) && (verboseLevel == 2)) cout<<endl;
378  }//for i
379 }//if RapConst
380 
381 // (used for the re-segmentation!) the idea here is that the transitions are dependent on the speaker time duration computed during the segmentation phase
382 if(transitionMethod=="SpeakerTime"){
383  unsigned long *speakerTime,timeTotal=0;
384  double transij,*speakerPercent;
385  speakerTime=new unsigned long[Server.getClusterCount()];
386  speakerPercent=new double[Server.getClusterCount()];
387  for(unsigned long icluster=0;icluster<Server.getClusterCount();icluster++){
388  //cluster extraction
389  SegCluster& cluster=Server.getCluster(icluster);
390  speakerTime[icluster]=totalFrame(cluster);
391  timeTotal+=speakerTime[icluster];
392  }//for icluster
393  for(unsigned long icluster=0;icluster<Server.getClusterCount();icluster++){
394  speakerPercent[icluster]=(double)speakerTime[icluster]/(double)timeTotal;
395  }//for icluster
396  for(unsigned long i=0;i<actualHMM.getNbState();i++){
397  for(unsigned long j=0;j<actualHMM.getNbState();j++){
398  if(i==j){
399  actualHMM.setTransition(gamma,i,j);
400  if (verbose) cout<<gamma<<" ";
401  }
402  else{
403  transij=(1-gamma)*(speakerPercent[j]+(speakerPercent[i]/Server.getClusterCount()));
404  actualHMM.setTransition(transij,i,j);
405  if (verbose) cout<<transij<<" ";
406  }
407  }//for j
408  if (verbose) cout<<endl;
409  }//for i
410 }//SpeakerTime
411 }//computeTransitions
