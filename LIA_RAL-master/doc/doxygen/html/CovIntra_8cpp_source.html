99 bool badSv(Matrix <double> & v) {
100  for (unsigned long j=0;j<v.cols();j++) {
101  if (abs(v(0,j))>myTINY) return false;
102  if (ISNAN(v(0,j)) || ISINF(v(0,j))) return true;
103  }
104 return true;

120 template <class T> class RealVectorStat {
121  private:
122  unsigned long _size;
123  unsigned long _nb;
124  RealVector <T> _xAcc;
125  RealVector <T> _xxAcc;
126  RealVector <double> _mean;
127  RealVector <double> _covInv;
128  public:
129  explicit RealVectorStat(unsigned long size) {
130  _size=size;
131  _nb=0;
132  _xAcc.setSize(_size);_xxAcc.setSize(_size);_mean.setSize(_size);_covInv.setSize(_size);
133  _xAcc.setAllValues(0);_xxAcc.setAllValues(0);_mean.setAllValues(0.0);_covInv.setAllValues(0.0);
134  }
135  ~ RealVectorStat() {};
136  void acc(RealVector <T> &v) {
137  if (_size!=v.size()) throw Exception("(RealVectorStat) size does not match!",__FILE__,__LINE__);
138  for (unsigned long i=0;i<v.size();i++) {
139  _xAcc[i]+=v[i];
140  _xxAcc[i]+=v[i]*v[i];
141  }
142  _nb++;
143  };
144  RealVector <double> & covInv() {
145  if (_nb==0) throw Exception("(RealVectorStat) Nothing accumulated!",__FILE__,__LINE__);
146  for (unsigned long i=0;i<_size;i++) {
147  _mean[i]=((double)_xAcc[i])/_nb;
148  _covInv[i]=1/((((double)_xxAcc[i])/_nb)-(_mean[i]*_mean[i]));
149  }
150  return _covInv;
151  };
152  RealVector <double> & mean() {
153  if (_nb==0) throw Exception("(RealVectorStat) Nothing accumulated!",__FILE__,__LINE__);
154  for (unsigned long i=0;i<_size;i++)
155  _mean[i]=((double)_xAcc[i])/_nb;
156  return _mean;
157  };
158 };

160 int CovIntra(Config & config) {

162  bool gmm=false;
163  if (config.existsParam("gmm")) {if(verbose) cout << "(CovIntra) Gmm mode: reading SV from gmm models"<<endl;gmm=true;}
164  bool svd=false;
165  if (config.existsParam("svd")) {if(verbose) cout << "(CovIntra) Computing the SVD"<<endl;svd=true;}
166  if (config.existsParam("klgmm")) {if(verbose) cout << "(CovIntra) Gmm KL mode: computing KL distance from gmm models"<<endl;gmm=true;}

168  XList fileList(config.getParam("ndx"),config);
169  XLine *pline;
170  String *pModel;
171  unsigned long svSize=config.getParam("vsize").toLong();
172  unsigned long nbSpeakers=fileList.getLineCount();
173  if (config.existsParam("nbSpeakers") && config.getParam("nbSpeakers").toLong() <= (long) nbSpeakers) nbSpeakers=config.getParam("nbSpeakers").toLong();
174 
175  // Compute desired Nb examples
176  unsigned long nbEx=0;
177  for (unsigned long s=0;s<nbSpeakers;s++)
178  nbEx+=fileList.getLine(s).getElementCount();
179  fileList.rewind();
180  Matrix <double> UBMsv;
181  loadMeanSv(config.getParam("inputWorldFilename"),UBMsv,config);
182  DMat CCt=svdNewDMat(svSize,nbEx); //svdlib format
183  RealVectorStat <double> acc(svSize);
184  if (verbose) cout<<"Channel Matrix dimension: "<<svSize<<","<<nbEx <<endl << "Speaker Matrix dimension: "<<svSize<<","<<nbSpeakers<<endl;
185  unsigned long idx=0;
186  for (unsigned long r=0;r<nbSpeakers;r++) {
187  pline=fileList.getLine();
188  if (verbose) cout << "Sp ["<<r<<"] " <<pline->getElementCount() << " sessions, " <<endl;
189  // Compute True Mean Statistic for Speaker in different sessions
190  Matrix <double> meanSv;
191  meanSv.setDimensions(1,svSize);
192  while((pModel=pline->getElement())!=NULL) {
193  if (verboseLevel > 1) cout << "Load " <<*pModel<<endl;
194  String filename;
195  Matrix <double> currSv;
196  if (!gmm) currSv.load(config.getParam("vectorFilesPath")+*pModel+config.getParam("vectorFilesExtension"),config);
197  else loadMeanSv(*pModel,currSv,config);
198  meanSv+=currSv;
199  }
200 
201  /************************ This compute InterSpeaker variability in a R vector **************************************************/
202  RealVector <double> vm;
203  for (unsigned long i=0;i<meanSv.cols();i++) {
204  double val=meanSv(0,i)/(pline->getElementCount());
205  val-=UBMsv(0,i);
206  vm.addValue(val);
207  }
208  acc.acc(vm);
209 
210  /************************* 2nd pass: Remove mean sv to get only channel (drop one) *******************************************/
211  for (unsigned long c=0;c<pline->getElementCount();c++) {
212  Matrix <double> currSv;
213  if (!gmm) currSv.load(config.getParam("vectorFilesPath")+pline->getElement(c)+config.getParam("vectorFilesExtension"),config);
214  else loadMeanSv(pline->getElement(c),currSv,config);
215  Matrix <double> meanMinusCurrent(meanSv); // if drop one to compute mean on the line
216  meanMinusCurrent-=currSv;
217  dividesSvByInteger(meanMinusCurrent,(pline->getElementCount()-1));
218  currSv-=meanMinusCurrent;
219  if (debug && badSv(currSv)) cout <<pline->getElement(c)<<" gives a bad channel supervector (zero|inf|nan value)"<<endl; // is faster as badSv is not evaluated if debug=0, at least that's what I think
220 
221  // Copy SV into matrix
222  for(unsigned long val=0;val<currSv.cols(); val++) {
223  CCt->value[val][idx]=currSv(0,val);
224  if(abs(CCt->value[val][idx])<myTINY) CCt->value[val][idx]=0.0; // sparsify matrix
225  }
226  idx++;
227  }
228  }
229  // Save interSp vector
230  RealVector <double>& interSp=acc.mean();
231  ((Matrix <double>)interSp).saveDT((String)"mean.inter",config);
232 
233  String filename=config.getParam("channelMatrix");
234  char *out=strdup(filename.c_str());
235  if (verbose && !svd) cout << "(CovIntra) Covariance matrix saved in " << config.getParam("channelMatrix")<< ", need to perform eigenvalue analysis yourself or add --svd flag to the config file" <<endl;
236  svdWriteDenseMatrix(CCt,out, SVD_F_DB);
237  free(out);
238  if (!svd) exit(1);

240  /****** SVDLIBC part: EigenValue decomposition with SVD ********************/
241  unsigned long nbIt=0;
242  double kappa=1e-6;
243  double las2end[2] = {-1.0e-30, 1.0e-30};
244  if (config.existsParam("nbIt")) nbIt=config.getParam("nbIt").toLong();

246  if (config.existsParam("bound")) {

253  for(int i=0;i<SCCt->vals;i++)
254  if (ISNAN(SCCt->value[i]) || ISINF(SCCt->value[i])) throw Exception("Error:nan or inf in sparse matrix file",__FILE__,__LINE__);
255  unsigned long nbEv=config.getParam("nbEigenVectors").toLong();
256  if (verbose) cout << "(CovIntra) Begin eigenValue problem resolution (using Doug Rohde's SVD C Library)" << endl;
257  SVDRec res=svdLAS2(SCCt,nbEv,nbIt,las2end,kappa);
258  svdFreeSMat(SCCt);
259  /*Save EigValues*/
260  Matrix <double> ev(1,res->d);
261  for (int i=0;i<res->d;i++) ev(0,i)=res->S[i];
262  String evFile=config.getParam("channelMatrix")+".S";
263  if (verbose) cout << "(CovIntra) Singluar Values (square root) in " <<evFile<< endl;
264  ev.saveDT(evFile,config);
265  /*Save EigVectors*/
266  Matrix <double> NAP(res->Ut->rows,res->Ut->cols);
267  if (verbose) cout<< "(CovIntra) Dimensions of U: ("<<NAP.rows()<<","<<NAP.cols()<<")"<< endl;
268  for (int i=0;i<res->Ut->rows;i++)
269  for (int j=0;j<res->Ut->cols;j++)
270  NAP(i,j)=res->Ut->value[i][j];
271  svdFreeSVDRec(res);
272  if (verbose) cout << "(CovIntra) U saved in " << config.getParam("channelMatrix")<<endl;
273  NAP.save(config.getParam("channelMatrix"),config);
274 }
275  catch (Exception& e) {cout << e.toString() << endl;}
276 return 0;
277 }
