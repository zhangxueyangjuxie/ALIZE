258 {
259  unsigned long vectSize = g1.getVectSize();
260  double a1=w1/(w1+w2);
261  double a2=1.0-a1;
262  for (unsigned long k=0;k<vectSize;k++)
263  {
264  double d=g1.getMean(k)-g2.getMean(k);
265  res.setCov(a1*g1.getCov(k)+a2*g2.getCov(k)+a1*a2*d*d,k);
266  res.setMean((a1*g1.getMean(k))+(a2*g2.getMean(k)),k);
267  }
268  res.computeAll();
269  w=w1+w2;
270 }

338 double likelihoodLoss(const DistribGD &g1,double w1,const DistribGD & g2,double w2)
339 {
340  unsigned long vectSize = g1.getVectSize();
341  double a1=w1/(w1+w2);
342  double a2=1.0-a1;
343  double d1=1.0;
344  double d2=1.0;
345  for (unsigned long k=0;k<vectSize;k++){
346  double d=g1.getMean(k)-g2.getMean(k);
347  double var=a1*g1.getCov(k)+a2*g2.getCov(k)+a1*a2*d*d;
348  d1*=var/g1.getCov(k);
349  d2*=var/g2.getCov(k);
350  }
351  return (0.5*(w1*log(d1)+w2*log(d2)));

783 //Compute the MLLR transformation for means of a input Model and store it in an output Model
784 
785 // remplacement de toutes les matrices carres par des Matrices rectangulaires
786 // Dimension de W (vectsize , vectsize+1)
787 // Dimension de Z (vectsize, vectsize+1)
788 // Dimension de G (vectsize +1 , vectsize +1) (en fait on a vectisze Matrices G : une par distribution)
789 
790 Matrix <double> computeMLLR (MixtureGD &inM,MixtureGD& outM,unsigned long frameCount, Config &config){
791  StatServer ss(config);
792 
793 
794 if(verbose)cout<<"MLLR begins"<<endl;
795  unsigned long distribCount=inM.getDistribCount(); // number of gaussians in the Mixture
796  unsigned long vectSize= inM.getVectSize(); // dimension of the gaussians
797  Matrix <double> W (vectSize, vectSize+1); // MMLR transformation matrix
798  Matrix <double> Z(vectSize, vectSize+1); // matrix (one unused colunn)
799  W.setAllValues(0);
800  Z.setAllValues(0);
801 
802  // Define and initialize the G matrices
803  DoubleSquareMatrix **G;
804  G=new DoubleSquareMatrix*[vectSize];
805  for (unsigned i=0;i<vectSize;i++) {
806  G[i]= new DoubleSquareMatrix();
807  G[i]->setSize(vectSize+1);
808  G[i]->setAllValues(0);
809  }
810  if(verboseLevel>=1 )cout<<"Initialization complete"<<endl<<"Begining the Statistic accumulation"<<endl;
811 
812  for (unsigned long j=0; j<distribCount;j++){
813  DistribGD & distri = inM.getDistrib(j);
814  DistribGD & distriMl = outM.getDistrib(j);
815  double occ=outM.weight(j)*frameCount;
816  DoubleVector mean(vectSize+1,1);
817  DoubleVector &cov=distri.getCovVect();
818 
819  //Initialization of the mean doubleVector (offset and other values)
820  mean[0]=1;
821  mean.addValue(distri.getMeanVect());
822 
823  // Compute the Z matrix
824  for (unsigned long p=0; p<vectSize;p++){
825  double occFrame=distriMl.getMean(p)*occ;
826  for (unsigned long q=0; q<vectSize+1;q++)
827  Z(p,q)+=occFrame*mean[q]/cov[p];
828  }
829  //Compute the G matrix
830  for (unsigned long p=0; p<vectSize;p++){
831  for (unsigned long q=0; q<vectSize+1; q++){
832  for (unsigned long r=0; r<vectSize+1; r++) (*G[p])(q,r) += occ*mean[q]*mean[r]/cov[p];
833  }
834  }
835  }
836 
837  if(verboseLevel >=1)cout<<"G and Z matrix calculated"<<endl;
838 
839  //Compute the W matrix using the G inverse and Z
840  for (unsigned long l=0; l<vectSize;l++) {
841  DoubleSquareMatrix G_inv(vectSize+1);
842  (*G[l]).invert(G_inv);
843  for (unsigned long c=0; c<vectSize+1;c++)
844  for( unsigned long k=0;k<vectSize+1;k++)
845  W(l,c) += G_inv(c,k)*Z(l,k);
846 
847  }
848  if(verboseLevel >= 1)cout<<"W complete"<<endl<<"Computing new means"<<endl;
849 
850  //Compute the mean vector of outModel
851  for (unsigned long j=0; j<distribCount;j++){
852  DoubleVector meanOut(vectSize, vectSize);
853  DistribGD & distri = inM.getDistrib(j);
854  DoubleVector &mean= distri.getMeanVect();
855  for (unsigned long i=0;i<vectSize;i++) meanOut[i]=W(i,0); //initialize the mean vector with the offset
856  for (unsigned long i=0;i<vectSize;i++) {
857  for (unsigned long k=0;k<vectSize;k++) meanOut[i]+=W(i,k+1)*mean[k];
858  }
859  outM.getDistrib(j).setMeanVect(meanOut); //update the mean vector of the out Mixture
860  }
861 
862  if(verbose) cout<<"MLLR finished, New Means computed"<<endl;
863 
864  copyVar(inM, outM);
865  copyWeight(inM,outM);
866 
867 return W;
868 }
