99  TVTranslate(XList &fileList) {
100  fileList.rewind(); XLine *pline;String *pFile;
101  unsigned long sent=0;
102  unsigned long loc=0;
103  while((pline=fileList.getLine())!=NULL) { // for each line of the NDX file
104  while((pFile=pline->getElement())!=NULL) { // for each file of the current line
105 
106  // if the file does not exist yet
107  if(_id.getIndex(*pFile) == -1){
108  _id.addElement(*pFile);
109 
110  ULongVector tmp(0,0);
111  _locindices.addObject(*new ULongVector(0,0));
112  _sessindices.addObject(*new ULongVector(0,0));
113  }
114  _sessionidx.addValue(sent);
115  _sessindices.getObject(_id.getIndex(*pFile)).addValue(sent);
116 
117  _locidx.addValue(loc);
118  _locindices.getObject(_id.getIndex(*pFile)).addValue(loc);
119 
120  sent++;
121  }
122  loc++;
123  }
124  };
