164  XLine *linep; // Pointor on the current test line

188 bool isNGram(short int *sym,NGram &tabS,unsigned long & tag){
189  bool find=false;
190  unsigned long idx;
191  for (idx=0;(!find) && (idx<tabS.getSize());idx++){
192  find=true;
193  for (unsigned long s=0;(find) && (s<tabS.getOrder());s++)
194  find=(sym[s]==tabS.getSymbol(idx,s));
195  }
196  if (find){
197  tag=idx;
198  return true;
199  }
200  else return false;
201 }

218 void computeLabelNGram(NGram & NG,SegCluster &cluster,SegCluster &clusterOut,ULongVector &tabS,unsigned long nbSym){
219  unsigned long begin[100]; // max order 100...
220  short int sym[100];
221  unsigned long end[100];
222  SegServer & segServerOut=clusterOut.getServer(); // Get the clusterserver reelated to the output
223  cluster.rewind();
224  Seg* seg; // Reset the reader at the begin of the input stream
225  while((seg=cluster.getSeg())!=NULL){ // For each of the selected segments
226  unsigned long idxFrame=seg->begin();
227  unsigned long endS=endSeg(seg);
228  if (endS>=nbSym) endS=nbSym; // Just if there is less symbol in the file than in the label
229  if (idxFrame>endS) idxFrame=endS;
230  unsigned long beginOOV=idxFrame;
231  bool oov=true;
232  if (debug) cout <<"begin Seg["<<idxFrame<<"]"<<endl;
233 
234  for (unsigned long n=0;(idxFrame<endS) &&(n<NG.getOrder()-1);n++){ // Recognize the (n-1) first symbols
235  begin[n]=idxFrame;
236  sym[n]=recognizeSymbol(idxFrame,endS,tabS);
237  end[n]=idxFrame-1;
238  if (debug) cout <<"sym ["<<sym[n]<<"] begin["<<begin[n]<<"] end["<<end[n]<<"] idxframe["<<idxFrame<<"]"<<endl;
239  }
240  while(idxFrame<endS){
241  begin[NG.getOrder()-1]=idxFrame;
242  sym[NG.getOrder()-1]=recognizeSymbol(idxFrame,endS,tabS);
243  end[NG.getOrder()-1]=idxFrame-1;
244  if (debug) cout <<"sym ["<<sym[NG.getOrder()-1]<<"] begin["<<begin[NG.getOrder()-1]
245  <<"] end["<<end[NG.getOrder()-1]<<"] idxframe["<<idxFrame<<"]"<<endl;
246  unsigned long tag;
247  if (isNGram(sym,NG,tag)){
248  if ((oov)&&(beginOOV<begin[0])){
249  if (debug) cout <<"OOV1 begin["<<beginOOV <<"] end["<<begin[0]-1<<"]"<<endl;
250  Seg &segTmp=segServerOut.createSeg(beginOOV,begin[0]-beginOOV,0,"oov",seg->sourceName());
251  clusterOut.add(segTmp);
252  }
253  if (debug) cout <<"NGRAM ["<<tag<<"] begin["<<begin[0] <<"] end["<<end[NG.getOrder()-1]<<"]"<<endl;
254  Seg &segTmp=segServerOut.createSeg(begin[0],end[NG.getOrder()-1]-begin[0]+1,0,String::valueOf(tag),seg->sourceName());
255  clusterOut.add(segTmp);
256  beginOOV=idxFrame;
257  oov=false;
258  }
259  else oov=true;
260  moveTab(begin,sym,end,NG.getOrder());
261  }
262  if (oov){
263  Seg &segTmp=segServerOut.createSeg(beginOOV,idxFrame-beginOOV,0,"oov",seg->sourceName());
264  clusterOut.add(segTmp);
265  if (debug) cout <<"OOV2 begin["<<beginOOV <<"] end["<<idxFrame-1<<"]"<<endl;
266  }
267  }
