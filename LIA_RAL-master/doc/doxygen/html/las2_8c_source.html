817 long lanczos_step(SMat A, long first, long last, double *wptr[],
818  double *alf, double *eta, double *oldeta,
819  double *bet, long *ll, long *enough, double *rnmp,
820  double *tolp, long n) {
821  double t, *mid, rnm = *rnmp, tol = *tolp, anorm;
822  long i, j;
823 
824  for (j=first; j<last; j++) {
825  mid = wptr[2];
826  wptr[2] = wptr[1];
827  wptr[1] = mid;
828  mid = wptr[3];
829  wptr[3] = wptr[4];
830  wptr[4] = mid;
831 
832  store(n, STORQ, j-1, wptr[2]);
833  if (j-1 < MAXLL) store(n, STORP, j-1, wptr[4]);
834  bet[j] = rnm;
835 
836  /* restart if invariant subspace is found */
837  if (!bet[j]) {
838  rnm = startv(A, wptr, j, n);
839  if (ierr) return j;
840  if (!rnm) *enough = TRUE;
841  }
842  if (*enough) {
843  /* added by Doug... */
844  /* These lines fix a bug that occurs with low-rank matrices */
845  mid = wptr[2];
846  wptr[2] = wptr[1];
847  wptr[1] = mid;
848  /* ...added by Doug */
849  break;
850  }
851 
852  /* take a lanczos step */
853  t = 1.0 / rnm;
854  svd_datx(n, t, wptr[0], 1, wptr[1], 1);
855  svd_dscal(n, t, wptr[3], 1);
856  svd_opb(A, wptr[3], wptr[0], OPBTemp);
857  svd_daxpy(n, -rnm, wptr[2], 1, wptr[0], 1);
858  alf[j] = svd_ddot(n, wptr[0], 1, wptr[3], 1);
859  svd_daxpy(n, -alf[j], wptr[1], 1, wptr[0], 1);
860 
861  /* orthogonalize against initial lanczos vectors */
862  if (j <= MAXLL && (fabs(alf[j-1]) > 4.0 * fabs(alf[j])))
863  *ll = j;
864  for (i=0; i < svd_imin(*ll, j-1); i++) {
865  store(n, RETRP, i, wptr[5]);
866  t = svd_ddot(n, wptr[5], 1, wptr[0], 1);
867  store(n, RETRQ, i, wptr[5]);
868  svd_daxpy(n, -t, wptr[5], 1, wptr[0], 1);
869  eta[i] = eps1;
870  oldeta[i] = eps1;
871  }
872 
873  /* extended local reorthogonalization */
874  t = svd_ddot(n, wptr[0], 1, wptr[4], 1);
875  svd_daxpy(n, -t, wptr[2], 1, wptr[0], 1);
876  if (bet[j] > 0.0) bet[j] = bet[j] + t;
877  t = svd_ddot(n, wptr[0], 1, wptr[3], 1);
878  svd_daxpy(n, -t, wptr[1], 1, wptr[0], 1);
879  alf[j] = alf[j] + t;
880  svd_dcopy(n, wptr[0], 1, wptr[4], 1);
881  rnm = sqrt(svd_ddot(n, wptr[0], 1, wptr[4], 1));
882  anorm = bet[j] + fabs(alf[j]) + rnm;
883  tol = reps * anorm;
884 
885  /* update the orthogonality bounds */
886  ortbnd(alf, eta, oldeta, bet, j, rnm);
887 
888  /* restore the orthogonality state when needed */
889  purge(n, *ll, wptr[0], wptr[1], wptr[4], wptr[3], wptr[5], eta, oldeta,
890  j, &rnm, tol);
891  if (rnm <= tol) rnm = 0.0;
892  }
893  *rnmp = rnm;
894  *tolp = tol;
895  return j;
896 }

1203  Description
1204  -----------

1207  a gap between them. The error bounds are then refined to reflect
1208  this.
1209 
1210 
1211  Arguments
1212  ---------
1213 
1214  (input)
1215  endl left end of interval containing unwanted eigenvalues
1216  endr right end of interval containing unwanted eigenvalues
1217  ritz array to store the ritz values

1223  --------------

1226  UTILITY svd_dmin
1227 
1228  ***********************************************************************/
1229 
1230 long error_bound(long *enough, double endl, double endr,
1231  double *ritz, double *bnd, long step, double tol) {
1232  long mid, i, neig;
1233  double gapl, gap;
1234 
1235  /* massage error bounds for very close ritz values */
1236  mid = svd_idamax(step + 1, bnd, 1);
1237 
1238  for (i=((step+1) + (step-1)) / 2; i >= mid + 1; i -= 1)
1239  if (fabs(ritz[i-1] - ritz[i]) < eps34 * fabs(ritz[i]))
1240  if (bnd[i] > tol && bnd[i-1] > tol) {
1241  bnd[i-1] = sqrt(bnd[i] * bnd[i] + bnd[i-1] * bnd[i-1]);
1242  bnd[i] = 0.0;
1243  }
1244 
1245 
1246  for (i=((step+1) - (step-1)) / 2; i <= mid - 1; i +=1 )
1247  if (fabs(ritz[i+1] - ritz[i]) < eps34 * fabs(ritz[i]))
1248  if (bnd[i] > tol && bnd[i+1] > tol) {
1249  bnd[i+1] = sqrt(bnd[i] * bnd[i] + bnd[i+1] * bnd[i+1]);
1250  bnd[i] = 0.0;
1251  }
1252 
1253  /* refine the error bounds */
1254  neig = 0;
1255  gapl = ritz[step] - ritz[0];
1256  for (i = 0; i <= step; i++) {
1257  gap = gapl;
1258  if (i < step) gapl = ritz[i+1] - ritz[i];
1259  gap = svd_dmin(gap, gapl);
1260  if (gap > bnd[i]) bnd[i] = bnd[i] * (bnd[i] / gap);
