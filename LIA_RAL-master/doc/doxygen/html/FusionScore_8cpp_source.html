70 void computeFusion(XLine & lineTab, XLine & finalScore, XList & weights,Histo &destH, Config & config)
71 {
72  String fusionMethod=config.getParam("fusionMethod");
73 
74  if (fusionMethod=="ArithMean")
75  {
76  double Score=0;
77  double sumWeight=0;
78  for (unsigned long i=0; i<lineTab.getElementCount(); i++)
79  {
80  cout << weights.getLine(0).getElement(i) << " * " << lineTab.getElement(i) << endl;
81  sumWeight += weights.getLine(0).getElement(i).toDouble();
82  Score+=weights.getLine(0).getElement(i).toDouble()*lineTab.getElement(i).toDouble();
83  }
84  Score/=sumWeight;
85  finalScore.addElement(String::valueOf(Score));
86  }
87  else if (fusionMethod=="GeoMean")
88  {
89  double Score=1;
90  double sumWeight=0;
91  double nbElm=lineTab.getElementCount();
92  for (int i=0; i<nbElm; i++){
93  sumWeight += weights.getLine(0).getElement(i).toDouble();
94  Score*=pow(weights.getLine(0).getElement(i).toDouble(),lineTab.getElement(i).toDouble());
95  }
96  if (Score < 0) {Score=pow(-1*Score,1.0/sumWeight); Score=-1*Score;}
97  else {Score=pow(Score,1.0/sumWeight);}
98  finalScore.addElement(String::valueOf(Score));
99  }
100  else if (fusionMethod=="multipleAryth"){
101  double Score=0;
102  // ici provoque warning de compilation convert from double to int
103  // long bin= destH(lineTab.getElement(0).toDouble(),1);
104  // modifié par éric -> introduction de cast to long pour supprimer warning
105  long bin= (long)destH(lineTab.getElement(0).toDouble(),1);
106 
107  if (debug) cout <<"score["<<lineTab.getElement(0).toDouble()<<"], fusion with line["<<bin<<"]"<<endl;
108  for (unsigned long i=0; i<lineTab.getElementCount(); i++)
109  {
110  Score+=weights.getLine(bin).getElement(i).toDouble()*lineTab.getElement(i).toDouble();
111  }
112  //Score/=lineTab.getElementCount(); give percentage in weight, it's better
113  finalScore.addElement(String::valueOf(Score));
114  }
115 
116  else {cerr <<"fusionMethod"<<fusionMethod<<"] unknown"<<endl; exit(1);}
117 }

120 void findNbest(XLine & tabScoreLine, int Nbest, int comp_double(const void *a, const void *b))
121 {
122  int nbElm=tabScoreLine.getElementCount();
123  double *tab = new double[nbElm];
124 
125  for (int i=0; i < nbElm; i++)
126  tab[i]=tabScoreLine.getElement(i).toDouble();
127 
128  qsort(tab,nbElm,sizeof(double),comp_double);
129 
130  tabScoreLine.reset();
131 
132  for (int j=nbElm-1; j >= nbElm-Nbest; j--)
133  tabScoreLine.addElement(String::valueOf(tab[j]));
134 }
