105 int EstimateDMatrix(Config & config){
106 
107  //Read the NDX file
108  String ndxFilename = config.getParam("ndxFilename");
109 
110  //Create and initialise the accumulator
111  JFAAcc jfaAcc(ndxFilename, config, "EstimateD");
112 
113  //Statistics
114  if((config.existsParam("loadAccs")) && config.getParam("loadAccs").toBool()){ //load pre-computed statistics
115  jfaAcc.loadN(config);
116  jfaAcc.loadN_h(config);
117  jfaAcc.loadF_X(config);
118  jfaAcc.loadF_X_h(config);
119  }
120  else{ //Compute statistics if they don't exists
121  jfaAcc.computeAndAccumulateJFAStat(config);
122  jfaAcc.saveAccs(config);
123  }
124 
125  //Initialise the D Matrix
126  if((config.existsParam("loadInitDMatrix"))&&(config.getParam("loadInitDMatrix").toBool())){ //Load the D matrix when existing
127  jfaAcc.loadD(config.getParam("initDMatrix"), config);
128  }
129  else{ //Initialise the D matrix randomly if does not exists
130  jfaAcc.initD(config);
131  }
132 
133  //Save the initial D matrix to be able restart the process with the same initialisation
134  if(config.existsParam("saveInitD") && config.getParam("saveInitD").toBool()){
135  String saveFilename = config.getParam("DMatrix") + "_init";
136  jfaAcc.saveD(saveFilename, config);
137  cout<<" (EstimateDMatrix) Save the initial D Matrix in "<<saveFilename<<endl;
138  }
139 
140  //Initialise the EV Matrix
141  if(config.existsParam("eigenVoiceMatrix")){ //Load the EV matrix when existing
142  jfaAcc.loadEV(config.getParam("eigenVoiceMatrix"), config);
143  }
144  else{ //Initialise the EV matrix to zero if does not exist
145  jfaAcc.getV().setAllValues(0.0);
146  cout<<"(EstimateDMatrix) Initialise NULL EigenVoice Matrix"<<endl;
147  }
148 
149  //Estimate Y factors for each speaker
150  jfaAcc.storeAccs();
151  jfaAcc.estimateVEVT(config);
152  jfaAcc.estimateAndInverseL_EV(config);
153  jfaAcc.substractMplusDZ(config);
154  jfaAcc.substractUX(config);
155  jfaAcc.estimateY(config);
156  //Reinitialise the accumulators
157  jfaAcc.restoreAccs();
158 
159  //Initialise the EC Matrix
160  if(config.existsParam("eigenChannelMatrix")){ //Load the EC matrix when existing
161  jfaAcc.loadEC(config.getParam("eigenChannelMatrix"), config);
162  }
163  else{ //Initialise the EC matrix to zero if does not exist
164  jfaAcc.getU().setAllValues(0.0);
165  cout<<" (EstimateDMatrix) Initialise NULL EigenChannel Matrix"<<endl;
166  }
167 
168  //Estimate X factors for each session
169  jfaAcc.storeAccs();
170  jfaAcc.estimateUEUT(config);
171  jfaAcc.estimateAndInverseL_EC(config);
172  jfaAcc.substractMplusVYplusDZ(config);
173  jfaAcc.estimateX(config);
174  //Reinitialise the accumulators
175  jfaAcc.restoreAccs();
176 
177  //iteratively retrain the D matrix
178  unsigned long nbIt = config.getParam("nbIt").toULong();
179 
180  jfaAcc.storeAccs();
181  for(unsigned long it=0; it<nbIt; it++){
182 
183  cout<<" (EstimateDMatrix) --------- start iteration "<<it<<" --------"<<endl;
184 
185  //Substract speaker statistics M + VY
186  jfaAcc.substractMplusVY(config);
187 
188  //Substract channel statistics UX
189  jfaAcc.substractUX(config);
190 
191  //Estimate Z for each speaker
192  jfaAcc.estimateZandD();
193 
194  //Reinitialise the accumulators
195  jfaAcc.resetTmpAcc();
196  jfaAcc.restoreAccs();
197 
198  //Save the D matrix at the end of the iteration
199  if(config.getParam("saveAllDMatrices").toBool()){
200  String s;
201  String output = config.getParam("DMatrix") + s.valueOf(it);
202  jfaAcc.saveD(output, config);
203  }
204  }

207  jfaAcc.saveD(config.getParam("DMatrix"), config);
208  cout<<" (EstimateDMatrix) --------- end of process --------"<<endl;
209 
210 return 0;
211 }
