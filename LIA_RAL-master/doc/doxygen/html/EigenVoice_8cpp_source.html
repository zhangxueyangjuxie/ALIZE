72 
73  //Read the NDX file

77  JFAAcc jfaAcc(ndxFilename, config,"EigenVoice");
78 
79  //Option used to check the Likelihood at each iteration
80  bool _checkLLK = false;
81  if (config.existsParam("checkLLK")) _checkLLK= config.getParam("checkLLK").toBool();
82  else if (verboseLevel >2) _checkLLK= true;
83 
84  //Statistics
85  if((config.existsParam("loadAccs")) && config.getParam("loadAccs").toBool()){ //load pre-computed statistics
86  cout<<" (EigenVoice)Load Accumulators"<<endl;
87  jfaAcc.loadN(config);
88  jfaAcc.loadN_h(config);
89  jfaAcc.loadF_X(config);
90  jfaAcc.loadF_X_h(config);
91  }
92  else{ //Compute statistics if they don't exists
93  jfaAcc.computeAndAccumulateJFAStat(config);
94  jfaAcc.saveAccs(config);
95  }
96 
97  //Initialise the EV Matrix
98  bool loadInitEigenVoiceMatrix = false;
99  if(config.existsParam("loadInitEigenVoiceMatrix")) loadInitEigenVoiceMatrix = config.getParam("loadInitEigenVoiceMatrix").toBool();
100  if(loadInitEigenVoiceMatrix){ //Load the EV matrix when existing
101  jfaAcc.loadEV(config.getParam("initEigenVoiceMatrix"), config);
102  }
103  else{ //Initialise the EV matrix randomly if does not exists
104  jfaAcc.initEV(config);
105  }
106 
107  //Save the initial V matrix to be able restart the process with the same initialisation
108  if(config.existsParam("saveInitEigenVoiceMatrix") && config.getParam("saveInitEigenVoiceMatrix").toBool()){
109  String initV = config.getParam("eigenVoiceMatrix")+"_init";
110  jfaAcc.saveV(initV, config);
111  cout<<" (EigenVoice) Save the initial EigenVoice Matrix in "<<initV<<endl;
112  }
113 
114  //iteratively retrain the EV matrix
115  unsigned long nbIt = config.getParam("nbIt").toULong();
116 
117  jfaAcc.storeAccs();
118  for(unsigned long it=0; it<nbIt; it++){
119 
120  cout<<" (EigenVoices) --------- start iteration "<<it<<" --------"<<endl;
121 
122  //On calcules les matrices vEvT
123  jfaAcc.estimateVEVT(config);
124 
125  //On calcules les matrices L et on les inverse (on integre la boucle sur tous les locuteurs dans cette fonction)
126  jfaAcc.estimateAndInverseL_EV(config);
127 
128  //On soustrait les statistiques du locuteur (M+DZ)*Ns pour chaque locuteur
129  jfaAcc.substractMplusDZ(config);
130 
131  //On soustrait pour chaque locuteur la composante canal de chaque session
132  jfaAcc.substractUX(config);
133 
134  //On update Y pour tous les locuteurs
135  jfaAcc.estimateYandV(config);
136 
137  if (_checkLLK) jfaAcc.verifyEMLK(config);
138 
139  //Update _V
140  jfaAcc.updateVestimate();
141 
142  //If the option is on, orthonormalize the matrix V
143  if(config.existsParam("orthonormalizeV") && (config.getParam("orthonormalizeV").toBool())){
144  if(verboseLevel > 0) cerr<<"Orthonormalize EV matrix"<<endl;
145  jfaAcc.orthonormalizeV();
146  }
147 
148  //Reinitialise the accumulators
149  jfaAcc.resetTmpAcc("EigenVoice");
150  jfaAcc.restoreAccs();
151 
152  //Save the V matrix at the end of the iteration
153  bool saveAllEVMatrices = false;
154  if(config.existsParam("saveAllEVMatrices")) saveAllEVMatrices=config.getParam("saveAllEVMatrices").toBool();
155  if(saveAllEVMatrices){
156  String s;
157  String output = config.getParam("eigenVoiceMatrix") + s.valueOf(it);
158  jfaAcc.saveV(output, config);
159  }
160  }
161 
162  cout<<" (EigenVoices) --------- save EigenVoices Matrix --------"<<endl;
163  jfaAcc.saveV(config.getParam("eigenVoiceMatrix"), config);
164  cout<<" (EigenVoices) --------- end of process --------"<<endl;
165 
166 return 0;
167 }
