74 /*******************************************
75  This is C default in SVM Light (inverse of average of train examples vectors norm)
76 *********************************************/
77 double getC(Matrix <double> train) {
78  double acc=0.0;
79  for (unsigned long i=0;i<train.rows();i++)
80  for (unsigned long j=0;j<train.cols();j++)
81  acc+=train(i,j)*train(i,j);
82  acc/=train.rows();
83 return 1/acc;
84 }

124 void readProblem(Matrix <double> & inputProb,RealVector <double>& labels,struct svm_problem & prob,struct svm_parameter& param,struct svm_node *x_space) {
125  if (verbose) cout << "(SVM) Reading SVM Classification pb" << endl;
126  int max_index=0;
127  int j=0;
128  int i=0;
129  for(i=0;i<prob.l;i++)
130  {
131  prob.x[i] = &x_space[j];
132  prob.y[i] = labels[i];
133  unsigned long k=0;
134  while(k<inputProb.cols())
135  {
136  (x_space[j].index)=k+1; //idx non sparse
137  (x_space[j].value)=inputProb(i,k); //val
138  ++j;
139  k++;
140  }
141  if(j>=1 && x_space[j-1].index > max_index)
142  max_index = x_space[j-1].index;
143  x_space[j++].index = -1;
144  }
145  if (verbose) cout << "(Svm) Problem in memory" << endl;
146 }

181 void loadHyperplane(String &filename,struct Svm_Hyperplane &h,Config & config) {
182  Matrix <double> model;
183  model.load(filename,config);
184  h.w.setSize(model.cols());
185  for (unsigned long i=0;i<model.cols()-1;i++)
186  h.w[i]=model(0,i);
187  h.offset=model(0,model.cols()-1);

189 // return hyperplane in a struct
190 void getHyperplane(struct Svm_Hyperplane &h,const svm_model *model,RealVector <double> &labels,Config & config) {
191  int libsvm_bug=1;
192  if(labels[0]!=1) libsvm_bug=-1;
193  int l = model->l;
194  const double * const *sv_coef = model->sv_coef;
195  const svm_node * const *SV = model->SV;
196  unsigned long vsize=config.getParam("vsize").toLong();
197  h.w.setSize(vsize);
198  h.w.setAllValues(0.0);
199  for(int i=0;i<l;i++) {
200  const svm_node *p = SV[i];
201  for (unsigned long j=0;j<h.w.size();j++,p++)
202  h.w[j]+=libsvm_bug*sv_coef[0][i]*p->value;
203  }
204  h.offset=libsvm_bug*model->rho[0];
205 }

274 int svmTrain(Config &config)
275 {
276  String outputFilename = config.getParam("outputFilename");
277  String vectorFilesPath = config.getParam("vectorFilesPath");
278  String modelFilesPath = config.getParam("modelFilesPath");
279  String vectorFilesExt = config.getParam("vectorFilesExtension");
280  String modelFilesExt = config.getParam("modelFilesExtension");
281  String inputFilename = config.getParam("inputFilename");
282  try{
283  Matrix <double> Bck;
284  if (verbose) cout << "(Svm) Using ["<<config.getParam("inputBckList")<<"] as additionnal exemples for training" << endl;
285  XLine impostorId=loadBck(Bck,config); // loading bck exemples
286  XList inputList;
287  bool tnorm=false;
288  if (config.existsParam("Tnorm")) {tnorm=true;if(verbose)cout <<"(Svm) Tnorm mode: Leave one out in inputBckList"<<endl;}
289  inputList.load(inputFilename,config);
290  XLine *pLine;
291 
292  while((pLine=inputList.getLine())!=NULL) {
293  String clientModelname=pLine->getElement(0);
294  String clientModelTrain=pLine->getElement(0); // same id
295  inputFilename=vectorFilesPath+clientModelTrain+vectorFilesExt;
296  outputFilename=modelFilesPath+clientModelname+modelFilesExt;
297  if (verbose) cout << "(Svm) Using ["<<inputFilename<<"] as training for: [" << outputFilename<<"]"<<endl;
298  Matrix <double>clientVector;
299  RealVector <double> labels;
300  if (!tnorm) {
301  clientVector.load(inputFilename,config);
302  unsigned long nbSessions=clientVector.rows();
303  if (verbose) cout << "(Svm) Number of sessions: " << nbSessions << endl; // todo
304  concat(clientVector,Bck);
305  labels.setSize(clientVector.rows());
306  for (unsigned long i=0;i<clientVector.rows();i++) {
307  if (i<nbSessions) labels[i]=1.0;
308  else labels[i]=-1.0;
309  }
310  }
312  else {
313  copy(Bck,clientVector);
314  labels.setSize(clientVector.rows());
315  for (unsigned long i=0;i<clientVector.rows();i++) { // only one session
316  if (clientModelTrain==impostorId.getElement(i)) {labels[i]=1.0;}
317  else labels[i]=-1.0;
318  }
319  }
320 
321  if (verbose) cout << "(Svm) Pb Dimensions: ("<<clientVector.rows()<<","<<clientVector.cols()<<")"<<endl;
322  //***************** Defines structures, and read Problem ****************************************//
323  struct svm_parameter param=definesParameter(config,clientVector);
324  struct svm_problem prob;
325  prob.l=clientVector.rows(); //number of exemples
326  prob.y=new double[prob.l];
327  prob.x=new svm_node*[prob.l];
328  struct svm_node *x_space=NULL;
329  x_space=new svm_node[(prob.l+1)*clientVector.cols()]; // the +1 was tricky to find!!!!!!!!
330  readProblem(clientVector,labels,prob,param,x_space); //alize way or reading problem
331  //************************************************************//
332 
333  const char* error_msg = svm_check_parameter(&prob,&param);
334  if(error_msg) {
335  fprintf(stderr,"Error: %s\n",error_msg); exit(1);
336  }
337  else {
338  if (verbose) cout<<"(Svm) Training Model"<<endl;
339  svm_model *model=svm_train(&prob,&param);
340  if (verboseLevel > 1) {
341  for (unsigned long i=0;i<(unsigned long)model->l;i++)
342  cout << "Sv["<<i<<"] ="<<model->sv_coef[0][i]<<endl;
343  }
344  struct Svm_Hyperplane h;
345  getHyperplane(h,model,labels,config); // only save w!
346  saveHyperplane(h,outputFilename,config);
347  if (verbose) cout<<"(Svm) Model saved in "<<outputFilename<<endl;
348  svm_destroy_model(model);
349  }
350  /* Destroy for each training */
351  svm_destroy_param(&param);
352  delete prob.y;
353  delete prob.x;
354  delete x_space;
355  }
356  }
357  catch (Exception& e){cout << e.toString().c_str() << endl;}
358 return 0;
359 }
