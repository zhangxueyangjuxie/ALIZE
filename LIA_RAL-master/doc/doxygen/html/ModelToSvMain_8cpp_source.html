58 void getMeanNorm(RealVector <double>& norm,MixtureGD & UBM) {
59  unsigned long mSize=UBM.getDistribCount();
60  unsigned long vSize=UBM.getVectSize();
61  norm.setSize(mSize*vSize);
62  for (unsigned long i=0;i<mSize;i++) {
63  DistribGD & d=UBM.getDistrib(i);
64  double w=UBM.weight(i);
65  for (unsigned long j=0;j<vSize;j++)
66  norm[i*vSize+j]=sqrt(w*d.getCovInv(j));
67  }
68 }

70 void getWeightNorm(RealVector <double>& norm,MixtureGD & UBM) {
71  unsigned long mSize=UBM.getDistribCount();
72  norm.setSize(mSize);
73  for (unsigned long i=0;i<mSize;i++)
74  norm[i]=1/sqrt(UBM.weight(i));
75 }
