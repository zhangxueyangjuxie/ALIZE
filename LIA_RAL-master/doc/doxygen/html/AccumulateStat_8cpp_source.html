170 static void *EMthread(void *threadarg) {
171  struct EMthread_data *my_data;
172  my_data = (struct EMthread_data *) threadarg;
173  SegCluster &selectedSegments=*(my_data->selectedSegments);
174  MixtureStat &emAcc=*(my_data->emAcc);
175  FeatureServer &fs=*(my_data->fs);
176  Config &config=*(my_data->config);
177  unsigned long nT=my_data->nThread;
178  double weight=1.0;
179  if (config.existsParam("weightedEM")) weight=config.getParam("weightedEM").toDouble();
180  // **************** Main loop
181  Seg* seg;
182  unsigned long cnt=0;
183  while(!stop) { // For each of the selected segments
184  // ***************** Reading features is locked
185  RefVector <Feature> featThreadBuff;
186  pthread_mutex_lock (&mutexsum);
187  if (stop) {pthread_mutex_unlock (&mutexsum);break;}
188  cnt++;
189  if ((seg=selectedSegments.getSeg())==NULL) {
190  pthread_mutex_unlock (&mutexsum);
191  if (verboseLevel >1) cout<<"Thread:"<<nT << " broke" << endl;
192  stop=true;
193  break;
194  }
195  unsigned long begin=seg->begin()+fs.getFirstFeatureIndexOfASource(seg->sourceName());
196  fs.seekFeature(begin);
197  for (unsigned long j=0;j<seg->length();j++){
198  featThreadBuff.addObject(*(new Feature),j);
199  fs.readFeature(featThreadBuff[j]);
200  }
201  pthread_mutex_unlock (&mutexsum);
202  // ***************** End lock
203 
204  // ***************** Accumulate EM on the frameTab
205  for (unsigned long j=0;j<seg->length();j++)
206  (*(my_data->llkAcc))+=log(emAcc.computeAndAccumulateEM(featThreadBuff[j],weight));
207  featThreadBuff.deleteAllObjects();
208  }
209  if (verboseLevel > 2) cout << "(AccumulateStatEM) Number of segments treated by thread ["<<nT<<"]="<<cnt<<endl;
210  pthread_exit((void*) 0);
211  return(void*)0;
212 }
