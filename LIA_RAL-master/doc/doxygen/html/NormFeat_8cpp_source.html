147 void computeWindowParam(Config &config,const DoubleVector &gMean,const DoubleVector &gCov,FeatureServer & fs,
148  SegCluster &window,double windowDuration,double windowComp,double frameLength,
149  DoubleVector &mean,DoubleVector &cov){
150  FrameAccGD frameAccu; // Defines a frame accumulator for mean and cov computation
151  frameAccu.reset();
152  accumulateStatFrame(frameAccu,fs, window, config);
153  mean = frameAccu.getMeanVect(); // Get the mean vector
154  cov = frameAccu.getStdVect(); // Get the std vector
155  if (debug) {cout <<"Compute param on"<<endl;showCluster(window);}
156  double duration=windowLength(window,frameLength);
157  if (verboseLevel>1) cout <<"Window real time["<<duration<<"]";
158  //if (duration>windowDuration) duration=windowDuration; ????????????????????????????????
159  double alpha=frameIdxToTime(totalFrame(window),frameLength);
160  if (verboseLevel>1) cout <<" data time["<<alpha <<"]";
161  alpha/=duration;
162  if (alpha>1) alpha =1.0;
163  if (verboseLevel>1) cout <<" Alpha["<<alpha<<"]";
164  alpha*=windowComp;
165  if (verboseLevel>1) cout <<" after Compensation["<<alpha<<"]"<<endl;;
166  for (unsigned long c=0;c<fs.getVectSize();c++){
167  mean[c]=(alpha*mean[c])+((1-alpha)*gMean[c]);
168  cov[c]=(alpha*cov[c]*cov[c]) +((1-alpha)*gCov[c]*gCov[c])
169  +(((1-alpha)*alpha)*pow(mean[c]-gMean[c],2));
170  cov[c]=sqrt(cov[c]);
171  }
