98  tvAcc.loadT(config.getParam("initTotalVariabilityMatrix"), config);

101  tvAcc.initT(config);
102  }
103 
104  //Save the initial Total Variability matrix to be able restart the process with the same initialisation
105  if(config.existsParam("saveInitTotalVariabilityMatrix") && config.getParam("saveInitTotalVariabilityMatrix").toBool()){
106  String initV = config.getParam("totalVariabilityMatrix")+"_init";

113  if(config.existsParam("minDivergence")) minDiv = config.getParam("minDivergence").toBool();
114 
115  //Iteratively retrain the EV matrix
116  unsigned long nbIt = config.getParam("nbIt").toULong();
117 
118  for(unsigned long it=0; it<nbIt; it++){
119 
120  cout<<" (TotalVariability) --------- start iteration "<<it<<" --------"<<endl;
121 
122  //Subtract mean from the statistics
123  tvAcc.substractM(config);
124 
125  //Compute TETt for each distribution
126  tvAcc.estimateTETt(config);
127

132  if (_checkLLK) tvAcc.verifyEMLK(config);
133

136

139  tvAcc.minDivergence();
140  }
141

143  if(config.existsParam("orthonormalizeT") && (config.getParam("orthonormalizeT").toBool())){
144  if(verboseLevel > 0) cout<<"Orthonormalize TV matrix"<<endl;
145  tvAcc.orthonormalizeT();
146  }
147 
148  //Reinitialize the accumulators
149  tvAcc.resetTmpAcc();
150 
151  //Reload statistics
152  tvAcc.loadN(config);
153  tvAcc.loadF_X(config);
154 
155  //Save the T matrix at the end of the iteration
156  bool saveAllEVMatrices = false;
157  if(config.existsParam("saveAllTVMatrices")) saveAllEVMatrices=config.getParam("saveAllTVMatrices").toBool();
158  if(saveAllEVMatrices){
159  String s;
160  String output = config.getParam("totalVariabilityMatrix") + s.valueOf(it);
161  tvAcc.saveT(output, config);
162 
163  // Save the new mean computed by Minimum Divergence if required
164  if(minDiv){
165  String sm;
166  String outputm = config.getParam("matrixFilesPath") + config.getParam("meanEstimate") + sm.valueOf(it) + config.getParam("saveMatrixFilesExtension");

168  }

171 
172  cout<<" (TotalVariability) --------- save Matrix --------"<<endl;
173  tvAcc.saveT(config.getParam("totalVariabilityMatrix"), config);
174  if(minDiv){
175  String outputm = config.getParam("matrixFilesPath") + config.getParam("meanEstimate") + config.getParam("saveMatrixFilesExtension");
176  ((Matrix<double>)tvAcc.getUbmMeans()).save(outputm, config);
177  }
178  cout<<" (TotalVariability) --------- end of process --------"<<endl;
179 
180  // If Required, output parameters for approximated i-vector extraction (ubmWeight or eigenDecomposition)
181  if(config.existsParam("approximationMode")){
182 
183  if(config.getParam("approximationMode") == "ubmWeight"){
184  cout<<" (TotalVariability) Compute weighted covariance matrix W for i-vector approximation using ubmWeight"<<endl;
185 
186  //Return the normalize T matrix
187  tvAcc.normTMatrix();
188  String normTFilename = config.getParam("totalVariabilityMatrix") + "_norm";
189  tvAcc.saveT(normTFilename, config);
190

193  MixtureServer ms(config);
194  MixtureGD& world = ms.loadMixtureGD(inputWorldFilename);
195  DoubleSquareMatrix W(tvAcc.getRankT());
196  W.setAllValues(0.0);

198  Matrix<double> tmpW(W);

200  tmpW.save(wFilename, config);
201  }
202  else if(config.getParam("approximationMode") == "eigenDecomposition"){
203  cout<<" (TotalVariability) Compute D and Q matrices for i-vector approximation using eigenDecomposition"<<endl;
204 
205  // Normalize matrix T
206  tvAcc.normTMatrix();
207  String normTFilename = config.getParam("totalVariabilityMatrix") + "_norm";
208  tvAcc.saveT(normTFilename, config);
209 
210  // Compute weighted co-variance matrix by using UBM weight coefficients
211  String inputWorldFilename = config.getParam("inputWorldFilename");
212  MixtureServer ms(config);
213  MixtureGD& world = ms.loadMixtureGD(inputWorldFilename);
214  DoubleSquareMatrix W(tvAcc.getRankT());
215  W.setAllValues(0.0);
216  tvAcc.getWeightedCov(W,world.getTabWeight(),config);
217 
218  // Eigen Decomposition of W to get Q
219  Matrix<double> tmpW(W);
220  Matrix<double> Q(tvAcc.getRankT(),tvAcc.getRankT());
221  Q.setAllValues(0.0);
222  tvAcc.computeEigenProblem(tmpW,Q,tvAcc.getRankT(),config);
223 
224  // Compute D matrices (approximation of Tc'Tc matrices)
225  Matrix<double> D(tvAcc.getNDistrib(),tvAcc.getRankT());

227  tvAcc.approximateTcTc(D,Q,config);
228 
229  config.getParam("matrixFilesPath") + config.getParam("totalVariabilityMatrix") + "_EigDec_D" + config.getParam("loadMatrixFilesExtension");
230 
231  String dFilename = config.getParam("matrixFilesPath") + config.getParam("totalVariabilityMatrix") + "_EigDec_D" + config.getParam("loadMatrixFilesExtension");
232  String qFilename = config.getParam("matrixFilesPath") + config.getParam("totalVariabilityMatrix") + "_EigDec_Q" + config.getParam("loadMatrixFilesExtension");
233  D.save(dFilename, config);
234  Q.save(qFilename, config);
235  }
236  else{
237  cout<<" (TotalVariability) This approximation mode does not exists"<<endl; // to check before to avoid being disapointed at the end
238  }
239 
240 
241  }
242 
243 
244 return 0;
