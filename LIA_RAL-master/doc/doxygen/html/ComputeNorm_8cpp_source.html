121 bool DistribNorm::computeMeanStd(double &mean,double &std,char mode, double percentH,double percentL){
122  if(tabScore.size() == 0) return false;
123  else{
124  unsigned long size=tabScore.size();
125  unsigned long begin=0;
126  unsigned long end=size;
127  if ((percentH!=0) || (percentL!=0)){
128  tabScore.descendingSort();
129  unsigned long discardH=(unsigned long) (((double) size) * percentH);
130  unsigned long discardL=(unsigned long) (((double) size) * percentL);
131  size-=(discardH+discardL);
132  begin=discardH;
133  end=tabScore.size()-discardL;
134  if (debug) cout<<"discardH["<<discardH<<"] discardL["<<discardL<<"] begin["<<begin<<"] end["<<end<<"]"<<endl;
135  }
136  double sum=0;
137  double sum2=0;
138  switch (mode){
139  case 0: //classical mean computation
140  for(unsigned int i=begin; i<end; i++){
141  sum += tabScore[i].lk;
142  sum2 += tabScore[i].lk * tabScore[i].lk;
143  }
144  mean=sum/(double)(size);
145  std=sqrt((sum2/(double)(size)-(mean*mean)));
146  break;
147  case 1: // Median instead of mean
148  mean=tabScore[begin+(size/2)].lk;
149  for(unsigned int i=begin; i<end; i++)
150  sum += abs(tabScore[i].lk-mean);
151  std=sum/(double)(size);
152  break;
153  default: cout <<"mean compute mode["<<mode<<"] unknown"<<endl;
154  exit(0);
155  }
156 
157  return true;
158  }
159 }
