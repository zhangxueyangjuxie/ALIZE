99  JFATranslate(XList &fileList) {
100  fileList.rewind(); XLine *pline;String *pFile;
101  unsigned long sent=0;
102  unsigned long loc=0;
103  while((pline=fileList.getLine())!=NULL) {
104  while((pFile=pline->getElement())!=NULL) {
105  _id.addElement(*pFile);
106  _locidx.addValue(loc);
107  _sessionidx.addValue(sent);
108  sent++;
109  }
110  loc++;
111  }
112  };
113  ~JFATranslate() {};

119  unsigned long locNb(const String &file) {
120  unsigned long idx=_idxOfID(file);
121  if (idx==-1) throw Exception("File not known",__FILE__,__LINE__);
122  return _locidx[idx];
123  }

129  unsigned long firstSession(unsigned long spk) {
130  unsigned long i;
131  for(i=0;;i++) if (_locidx[i]==spk) break;
132  return (_sessionidx[i]);
133  }

140  unsigned long numSession(unsigned long spk, unsigned long session) {
141  unsigned long i, cnt;
142  if(spk==0) return(session);
143  for(i=session-1,cnt=0;;i--,cnt++) if (_locidx[i]!=spk) break;
144  return (cnt);
145  }
