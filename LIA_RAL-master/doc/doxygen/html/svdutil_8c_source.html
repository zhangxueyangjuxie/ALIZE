564 double svd_random2(long *iy) {

566  static long ia, ic, mic;
567  static double halfm, s;
568 
569  /* If first entry, compute (max int) / 2 */
570  if (!m2) {
571  m2 = 1 << (8 * (int)sizeof(int) - 2);
572  halfm = m2;
573 
574  /* compute multiplier and increment for linear congruential
575  * method */
576  ia = 8 * (long)(halfm * atan(1.0) / 8.0) + 5;
577  ic = 2 * (long)(halfm * (0.5 - sqrt(3.0)/6.0)) + 1;
578  mic = (m2-ic) + m2;
579 
580  /* s is the scale factor for converting to floating point */
581  s = 0.5 / halfm;
582  }
583 
584  /* compute next random number */
585  *iy = *iy * ia;
586 
587  /* for computers which do not allow integer overflow on addition */
588  if (*iy > mic) *iy = (*iy - m2) - m2;
589 
590  *iy = *iy + ic;
591 
592  /* for computers whose word length for addition is greater than
593  * for multiplication */
594  if (*iy / 2 > m2) *iy = (*iy - m2) - m2;
595 
596  /* for computers whose integer overflow affects the sign bit */
597  if (*iy < 0) *iy = (*iy + m2) + m2;
598
