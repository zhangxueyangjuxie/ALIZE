72 int InfoTarget(Config& config)
73 {
74  String inputClientListFileName = config.getParam("targetIdList");
75  bool fixedLabelSelectedFrame;
76  String labelSelectedFrames;
77  if (config.existsParam("useIdForSelectedFrame")) // the ID of each speaker is used as labelSelectedFrame
78  fixedLabelSelectedFrame=false;
79  else{ // the label is decided by the command line and is unique for the run
80  labelSelectedFrames=config.getParam("labelSelectedFrames");
81  if (verbose) cout << "Computing on" << labelSelectedFrames << " label" << endl;
82  fixedLabelSelectedFrame=true;
83  }
84  unsigned long maxFrame=config.getParam("maxFrame").toLong();
85  String outputFilename=config.getParam("outputFilename");
86 
87 
88  ofstream outputFile(outputFilename.c_str(),ios::out| ios::trunc);
89  try{
90  XList inputClientList(inputClientListFileName,config); // read the Id + filenames for each client
91  XLine * linep;
92  if (verbose) cout << "InfoTarget" << endl;
93  // *********** Target loop *****************
94  while ((linep=inputClientList.getLine()) != NULL){ // linep gives the XLine with the Id of a given client and the list of files
95  String *id=linep->getElement(); // Get the Client ID (id)
96  outputFile<<*id;
97  String currentFile="";
98  XLine featureFileListp=linep->getElements(); // Get the list of feature file for the client (end of the line)
99  if (verbose) cout << "Info model ["<<*id<<"]"<<endl;
100  if (!fixedLabelSelectedFrame){ // the ID is used as label for selecting the frame
101  labelSelectedFrames=*id;
102  if (debug) cout <<*id<<" is used for label selected frames"<<endl;
103  }
104  // label files reading - It creates, for each file and each label, a cluster of segments - will be integrated witth the featre s - asap
105  SegServer segmentsServer; // Reading the segmentation files for each feature input file
106  LabelServer labelServer;
107  initializeClusters(featureFileListp,segmentsServer,labelServer,config); // Reading the segmentation files for each feature input file
108  unsigned long codeSelectedFrame=labelServer.getLabelIndexByString(labelSelectedFrames); // Get the index of the cluster with in interest audio segments
109  SegCluster& selectedSegments=segmentsServer.getCluster(codeSelectedFrame); // Gives the cluster of the selected/used segments
110  Seg *seg; // Will give the current segment
111  unsigned long frameCount=0;
112  selectedSegments.rewind(); // at the begin of the selected segments list
113  while(((seg=selectedSegments.getSeg())!=NULL) && (frameCount<maxFrame)){ // For each of the selected segments until the amount of data is get
114  frameCount+=seg->length();
115  cout << seg->sourceName()<<" "<<seg->begin()<<" "<<seg->length()<<" Total time="<<frameCount<<endl;
116  if (seg->sourceName()!=currentFile){
117  outputFile<<" "<<seg->sourceName();
118  currentFile=seg->sourceName();
119  }
120  } // end of the initial Train Iteration loop
121  outputFile<<endl;
122  if (verbose) cout << "Save info client ["<<*id<<"]" << endl;
123  } // end of the the target loop
124  } // fin try
125 
126  catch (Exception& e)
127  {
128  cout << e.toString().c_str() << endl;
129  }
130  return 0;
131 }

135 {

138  // label for selected frames - Only the frames associated with this label, in the label files, will be used
139  bool fixedLabelSelectedFrame=true;
140  String labelSelectedFrames;
141  if (config.existsParam("useIdForSelectedFrame")) // the ID of each speaker is used as labelSelectedFrame ?
142  fixedLabelSelectedFrame=(config.getParam("useIdForSelectedFrame").toBool()==false);
143  if (fixedLabelSelectedFrame) // the label is decided by the command line and is unique for the run
144  labelSelectedFrames=config.getParam("labelSelectedFrames");
145 
146 try{
147  MixtureServer ms(config);
148  if (verbose) cout << "(IvExtractor) TotalVariability - Load world model [" << inputWorldFilename<<"]"<<endl;
149  MixtureGD& world = ms.loadMixtureGD(inputWorldFilename);
150 
151  //Load the statistics from files or compute statistics for all segments at once
152  //Read the NDX file
153  String ndxFilename = config.getParam("targetIdList");
154 
155  //Remove the first element of each line which is the model name
156  XList tmpFileList(ndxFilename);
157  XList fileList;
158  for(unsigned long ll=0;ll<tmpFileList.getLineCount();ll++){
159  fileList.addLine();
160  for(unsigned long i=1;i<tmpFileList.getLine()->getElementCount();i++){
161  fileList.getLine(fileList.getLineCount()-1).addElement(tmpFileList.getLine(ll).getElement(i));
162  }
163  }
164 
165  //Create and initialise the accumulator
166  TVAcc tvAcc(fileList, config);
167 
168  // Load TotalVariability Matrix
169  tvAcc.loadT(config.getParam("totalVariabilityMatrix"),config);
170 
171 
172  //Statistics
173  if((config.existsParam("loadAccs")) && config.getParam("loadAccs").toBool()){ //load pre-computed statistics
174  cout<<" (IvExtractor) Load Accumulators"<<endl;
175  tvAcc.loadN(config);
176  tvAcc.loadF_X(config);
177  }
178  else{ //Compute statistics if they don't exists
179  tvAcc.computeAndAccumulateTVStat(config);
180  tvAcc.saveAccs(config);
181  }
182 
183  // Then load the meanEstimate computed by minDiv if required
184  DoubleVector meanEstimate = tvAcc.getUbmMeans();
185  if(config.existsParam("minDivergence")&& config.getParam("minDivergence").toBool()){
186  String minDivName = config.getParam("matrixFilesPath") + config.getParam("meanEstimate") + config.getParam("loadMatrixFilesExtension");
187  Matrix<double> tmpMean(minDivName,config);
188  for(unsigned long i=0;i<meanEstimate.size();i++){
189  meanEstimate[i] = tmpMean(0,i);
190  }
191  }
192  //Update the mean Estimate
193  cout<<" (IvExtractor) Load Mean Estimate"<<endl;
194  tvAcc.loadMeanEstimate(meanEstimate);
195 
196  //Substract mean from the statistics
197  tvAcc.substractM(config);
198 
199  //Compute vEvT for each session
200  tvAcc.estimateTETt(config);
201 
202  // Estimate I-Vectors
203  tvAcc.estimateW(config);
204 
205  cout<<"--------- save IV by File --------"<<endl;
206  tvAcc.saveWbyFile(config);
207  cout<<"--------- end of process --------"<<endl;
208 
209 } // fin try
210 catch (Exception& e) {cout << e.toString().c_str() << endl;}
