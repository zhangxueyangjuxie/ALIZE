64 // For a single frame
65 void computeExpansion(Feature & f,Feature & expF) {
66  expF[0]=1;// first value in expansion is one
67  unsigned long size=f.getVectSize();
68  for (unsigned long i=0;i<size;i++) // first [1 VectSize+1] values in the expension are the feature
69  expF[i+1]=f[i];
70 
71  unsigned long idx=0; // index of the expansion
72  //calculate expansion
73  for (unsigned long i=0;i<size+1;i++){
74  for (unsigned long j=i;j<size+1;j++){
75  for (unsigned long k=j;k<size+1;k++){
76  expF[idx] = expF[i]*expF[j]*expF[k] ;// all combinations with repetion
77  idx++;
78  }
79  }
80  }
81 
82 }

85 void computeAndAccumulateExpansion(FeatureServer & fs,FrameAccGD & avgExp,unsigned long idxBeginFrame,unsigned long nbFrames,Config & config) {
86 
87  fs.seekFeature(idxBeginFrame); // go to the frame in the buffer (and load it if needed)
88  unsigned long vectSize=fs.getVectSize();
89  unsigned long expSize = (vectSize + 3) * (vectSize + 2) * (vectSize + 1) / 6 ;
90  if (debug) cout << "Expension size: "<<expSize<<endl;
91  Feature expF(expSize);
92  for (unsigned long n=0;n<nbFrames;n++){
93  Feature f;
94  fs.readFeature(f);
95  // Defines size of expansion
96  computeExpansion(f,expF);
97  avgExp.accumulate(expF);
98  }
99 }

164 int PolyExpand(Config & config){
165  try {
166  XList inputList(config.getParam("inputFeatureFilename"),config);
167  XLine * pLine;
168  bool computeR=config.existsParam("computeR");
169  bool normalize=config.existsParam("normalize");
170  String labelSelectedFrames=config.getParam("labelSelectedFrames");
171  bool verbose=false;
172  if (config.existsParam("verbose")) verbose=config.getParam("verbose").toBool();
173  XList xMat;
174  if (normalize) {
175  xMat.load(config.getParam("normalize"),config);
176  }
177 // Accumulate avg expansion over multiple files
178  FrameAccGD avgExp;
179  while((pLine=inputList.getLine())!=NULL) {
180  String filename=pLine->getElement(0);
181  if (verbose) cout << "Processing file: ["<<filename<<"] ... ";
182  FeatureServer fs(config,filename);
183  SegServer segmentsServer; // Create the segment server for managing the segments/clusters
184  LabelServer labelServer; // Create the lable server, for indexing the segments/clusters
185  initializeClusters(filename,segmentsServer,labelServer,config); // Reading the segmentation files for each feature input file
186  verifyClusterFile(segmentsServer,fs,config); // Verify if the segments ending before the end of the feature files...
187  unsigned long codeSelectedFrame=labelServer.getLabelIndexByString(labelSelectedFrames);// Get the index of the cluster with in interest audio segments
188  SegCluster& selectedSegments=segmentsServer.getCluster(codeSelectedFrame); // Gives the cluster of the selected/used segments
189  // Accumulates average poly expansion
190  computeAndAccumulateExpansion(fs,avgExp,selectedSegments,config);
191  if (verbose) cout << "Done" <<endl;
192  // get mean exp
193  if (!computeR) {
194  DoubleVector avgExpVect=avgExp.getMeanVect();
195  if (normalize) multiplyByR(avgExpVect,xMat);
196  String outFile=config.getParam("vectorFilesPath")+filename+config.getParam("vectorFilesExtension");
197  outputInstance(avgExpVect,outFile,config);
198  avgExp.reset();
199  }
200  }
201  if (computeR) {
202  const DoubleVector & meanR=avgExp.getMeanVect();
203  DoubleVector R=avgExp.getxAccVect();
204  computeRSqrt(R,avgExp.getCount());
205  // output R matrix;
206  outputR(R,meanR,config);
207  }
208 }
209  catch (Exception& e) {cout << e.toString() << endl;}
210 return 0;
211 }
