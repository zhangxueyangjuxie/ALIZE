79  long _idxOfID(const String &file) {
80  _id.rewind();
81  for(unsigned long i=0;i<_id.getElementCount();i++)
82  if(file==_id.getElement(i)) return i;
83  return -1;
84  }
85 
86  public:
87  Translate() {};
88 
90  Translate(XList &fileList) {
91  fileList.rewind(); XLine *pline;String *pFile;
92  unsigned long sent=0;
93  unsigned long loc=0;
94  while((pline=fileList.getLine())!=NULL) {
95  while((pFile=pline->getElement())!=NULL) {
96  _id.addElement(*pFile);
97  _locidx.addValue(loc);
98  _sessionidx.addValue(sent);
99  sent++;
100  }
101  loc++;
102  }
103  };
104  ~Translate() {};
105 
107  unsigned long locNb(const String &file) {
108  long idx=_idxOfID(file);
109  if (idx==-1) throw Exception("File not known",__FILE__,__LINE__);
110  return _locidx[idx];
111  }
112 
114  unsigned long sessionNb(const String &file) {
115  long idx=_idxOfID(file);
116  if (idx==-1) throw Exception("File not known",__FILE__,__LINE__);
117  return _sessionidx[idx];
118  }
